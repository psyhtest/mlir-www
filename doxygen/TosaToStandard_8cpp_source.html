53 // This converts the TOSA ApplyScale operator to a set of StandardOps ops,
54 // using 64-bit operations to perform the necessary multiply, bias, and shift.
55 // Multiple types are used to use minimal bit width operations.
56 class ApplyScaleOpConverter : public OpRewritePattern<tosa::ApplyScaleOp> {
57 public:
58  using OpRewritePattern<tosa::ApplyScaleOp>::OpRewritePattern;
59 
60  LogicalResult matchAndRewrite(tosa::ApplyScaleOp op,
61  PatternRewriter &rewriter) const final {
62  Location loc = op.getLoc();
63  Value value32 = op.value();
64  Value multiplier32 = op.multiplier();
65  Value shift8 = op.shift();
66  bool doubleRound = op.double_round();
67 
68  Value one8 = rewriter.create<ConstantOp>(
69  loc, rewriter.getIntegerAttr(rewriter.getIntegerType(8), 1));
70  Value one32 = rewriter.create<ConstantOp>(
71  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 1));
72  Value one64 = rewriter.create<ConstantOp>(
73  loc, rewriter.getIntegerAttr(rewriter.getI64Type(), 1));
74 
75  Value shiftSubOne8 = rewriter.create<SubIOp>(loc, shift8, one8);
76 
77  // The rounding value semantics below equate to the following code:
78  // int64_t round = 1 << (shift - 1);
79  // if (double_round) {
80  // if (shift > 31 && value >= 0) round += 1<<30;
81  // if (shift > 31 && value < 0) round -= 1<<30;
82  // }
83  //
84  // Note that minimal bitwidth operators are used throughout the block.
85 
86  Value shift32 = rewriter.create<mlir::SignExtendIOp>(
87  loc, rewriter.getI32Type(), shift8);
88 
89  Value round64 = rewriter.create<mlir::ShiftLeftOp>(
90  loc, one64,
91  rewriter.create<SignExtendIOp>(loc, rewriter.getI64Type(),
92  shiftSubOne8));
93 
94  // Double rounding is performing a round operation before the shift
95  if (doubleRound) {
96  Value zero32 = rewriter.create<ConstantOp>(
97  loc, rewriter.getZeroAttr(rewriter.getI32Type()));
98  Value thirty32 = rewriter.create<ConstantOp>(
99  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 30));
100 
101  Value shiftThirty32 =
102  rewriter.create<mlir::ShiftLeftOp>(loc, one32, thirty32);
103  Value shiftThirty64 = rewriter.create<mlir::SignExtendIOp>(
104  loc, rewriter.getI64Type(), shiftThirty32);
105 
106  // Round value needs to with be added or sbustracted depending on
107  Value roundAdd64 =
108  rewriter.create<mlir::AddIOp>(loc, round64, shiftThirty64);
109  Value roundSub64 =
110  rewriter.create<mlir::SubIOp>(loc, round64, shiftThirty64);
111 
112  Value valueGreaterThanZero = rewriter.create<mlir::CmpIOp>(
113  loc, CmpIPredicate::sge, value32, zero32);
114 
115  Value doubleRound64 = rewriter.create<mlir::SelectOp>(
116  loc, valueGreaterThanZero, roundAdd64, roundSub64);
117 
118  // We only perform double rounding if the shift value is greater than 32.
119  Value thirtyTwo32 = rewriter.create<ConstantOp>(
120  loc, rewriter.getIntegerAttr(rewriter.getI32Type(), 32));
121  Value shiftGreaterThanThirtyTwo = rewriter.create<mlir::CmpIOp>(
122  loc, CmpIPredicate::sge, shift32, thirtyTwo32);
123  round64 = rewriter.create<mlir::SelectOp>(loc, shiftGreaterThanThirtyTwo,
124  doubleRound64, round64);
125  }
126 
127  // The computation below equates to the following pseudocode:
128  // int64_t result = (int64_t)value * multiplier + round;
129  // result = result >> shift;
130  //
131  // Note that multiply and shift need to be perform in i64 to preserve bits.
132 
133  Value value64 =
134  rewriter.create<SignExtendIOp>(loc, rewriter.getI64Type(), value32);
135  Value multiplier64 = rewriter.create<SignExtendIOp>(
136  loc, rewriter.getI64Type(), multiplier32);
137  Value shift64 =
138  rewriter.create<SignExtendIOp>(loc, rewriter.getI64Type(), shift8);
139 
140  // Multiply as a pair of i64 values to guarantee the end value fits.
141  Value result64 = rewriter.create<MulIOp>(loc, value64, multiplier64);
142  result64 = rewriter.create<AddIOp>(loc, result64, round64);
143  result64 =
144  rewriter.create<mlir::SignedShiftRightOp>(loc, result64, shift64);
145 
146  Value result32 = rewriter.create<mlir::TruncateIOp>(
147  loc, rewriter.getI32Type(), result64);
148 
149  rewriter.replaceOp(op, result32);
150  return success();
151  }
152 };
