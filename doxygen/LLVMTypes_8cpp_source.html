171 LLVMStructType LLVMStructType::getNewIdentified(MLIRContext *context,
172  StringRef name,
173  ArrayRef<Type> elements,
174  bool isPacked) {
175  std::string stringName = name.str();
176  unsigned counter = 0;
177  do {
178  auto type = LLVMStructType::getIdentified(context, stringName);
179  if (type.isInitialized() || failed(type.setBody(elements, isPacked))) {
180  counter += 1;
181  stringName = (Twine(name) + "." + std::to_string(counter)).str();
182  continue;
183  }
184  return type;
185  } while (true);
186 }

340 bool mlir::LLVM::isCompatibleType(Type type) {
341  // Only signless integers are compatible.
342  if (auto intType = type.dyn_cast<IntegerType>())
343  return intType.isSignless();
344 
345  // 1D vector types are compatible if their element types are.
346  if (auto vecType = type.dyn_cast<VectorType>())
347  return vecType.getRank() == 1 && isCompatibleType(vecType.getElementType());
348 
349  // clang-format off
350  return type.isa<
351  BFloat16Type,
352  Float16Type,
353  Float32Type,
354  Float64Type,
355  Float80Type,
356  Float128Type,
357  LLVMArrayType,
358  LLVMFunctionType,
359  LLVMLabelType,
360  LLVMMetadataType,
361  LLVMPPCFP128Type,
362  LLVMPointerType,
363  LLVMStructType,
364  LLVMTokenType,
365  LLVMFixedVectorType,
366  LLVMScalableVectorType,
367  LLVMVoidType,
368  LLVMX86MMXType
369  >();
370  // clang-format on
371 }

378 bool mlir::LLVM::isCompatibleVectorType(Type type) {
379  if (type.isa<LLVMFixedVectorType, LLVMScalableVectorType>())
380  return true;
381 
382  if (auto vecType = type.dyn_cast<VectorType>()) {
383  if (vecType.getRank() != 1)
384  return false;
385  Type elementType = vecType.getElementType();
386  if (auto intType = elementType.dyn_cast<IntegerType>())
387  return intType.isSignless();
388  return elementType.isa<BFloat16Type, Float16Type, Float32Type, Float64Type,
389  Float80Type, Float128Type>();
390  }
391  return false;
392 }

427 llvm::TypeSize mlir::LLVM::getPrimitiveTypeSizeInBits(Type type) {
428  assert(isCompatibleType(type) &&
429  "expected a type compatible with the LLVM dialect");
430 
431  return llvm::TypeSwitch<Type, llvm::TypeSize>(type)
432  .Case<BFloat16Type, Float16Type>(
433  [](Type) { return llvm::TypeSize::Fixed(16); })
434  .Case<Float32Type>([](Type) { return llvm::TypeSize::Fixed(32); })
435  .Case<Float64Type, LLVMX86MMXType>(
436  [](Type) { return llvm::TypeSize::Fixed(64); })
437  .Case<Float80Type>([](Type) { return llvm::TypeSize::Fixed(80); })
438  .Case<Float128Type>([](Type) { return llvm::TypeSize::Fixed(128); })
439  .Case<IntegerType>([](IntegerType intTy) {
440  return llvm::TypeSize::Fixed(intTy.getWidth());
441  })
442  .Case<LLVMPPCFP128Type>([](Type) { return llvm::TypeSize::Fixed(128); })
443  .Case<LLVMFixedVectorType>([](LLVMFixedVectorType t) {
444  llvm::TypeSize elementSize =
445  getPrimitiveTypeSizeInBits(t.getElementType());
446  return llvm::TypeSize(elementSize.getFixedSize() * t.getNumElements(),
447  elementSize.isScalable());
448  })
449  .Case<VectorType>([](VectorType t) {
450  assert(isCompatibleVectorType(t) &&
451  "unexpected incompatible with LLVM vector type");
452  llvm::TypeSize elementSize =
453  getPrimitiveTypeSizeInBits(t.getElementType());
454  return llvm::TypeSize(elementSize.getFixedSize() * t.getNumElements(),
455  elementSize.isScalable());
456  })
457  .Default([](Type ty) {
458  assert((ty.isa<LLVMVoidType, LLVMLabelType, LLVMMetadataType,
459  LLVMTokenType, LLVMStructType, LLVMArrayType,
460  LLVMPointerType, LLVMFunctionType>()) &&
461  "unexpected missing support for primitive type");
462  return llvm::TypeSize::Fixed(0);
463  });
464 }
