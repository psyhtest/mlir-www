4 """DSL for constructing affine expressions and maps.
5 
6 These python wrappers allow construction of affine expressions in a more
7 pythonic fashion that is later instantiated as an IR AffineExpr. Separating the
8 AST from construction of the map allows for manipulations of symbols and dims
9 beyond the scope of one expression.
10 
11 Affine expression construction:
12  >>> with _ir.Context():
13  ... s = AffineBuildState()
14  ... (S.K + S.M).build(s)
15  ... (S.K * S.M).build(s)
16  ... (S.K // S.M).build(s)
17  ... (S.K / S.M).build(s)
18  ... (S.K % 4).build(s)
19  ... (D.i + D.j * 4).build(s)
20  ... s
21  AffineExpr(s0 + s1)
22  AffineExpr(s0 * s1)
23  AffineExpr(s0 floordiv s1)
24  AffineExpr(s0 ceildiv s1)
25  AffineExpr(s0 mod 4)
26  AffineExpr(d0 + d1 * 4)
27  AffineBuildState<
28  symbols={'K': 0, 'M': 1}
29  dims={'i': 0, 'j': 1}>
30 
31 In the DSL, dimensions and symbols are name-uniqued instances of DimDef and
32 SymbolDef. There are shortcut "expando" instances that will create a
33 corresponding DimDef/SymbolDef upon accessing an attribute:
34 
35 Referencing a named dimension:
36 
37  >>> D.i
38  Dim(i)
39  >>> D.a is D.b
40  False
41  >>> D.a is D.a
42  True
43 
44 Referencing a named symbol:
45 
46  >>> S.foobar
47  Symbol(foobar)
48  >>> S.a is S.b
49  False
50  >>> S.a is S.a
51  True
52 """

71 class AffineBuildState:
72  """Internal state for the AffineExprDef._create impls.
73 
74  Note that a "local" AffineBuildState can be created relative to a "global"
75  AffineBuildState. In that case, any affine expressions built will inherit
76  symbol and dim bindings from the global state and will update both as new
77  ones are discovered. This allows for building expressions across contexts
78  which share a common symbol and dim space.
79  """
80 
81  def __init__(self,
82  *,
83  global_state: "AffineBuildState" = None,
84  allow_new_symbols: bool = True,
85  allow_new_dims: bool = True):
86  if not global_state:
87  self.all_symbols = dict() # type: Dict[str, int]
88  self.all_dims = dict() # type: Dict[str, int]
89  else:
90  # Alias the global dict.
91  self.all_symbols = global_state.all_symbols
92  self.all_dims = global_state.all_dims
93 
94  # Map of symbols and dims in the current build.
95  self.local_symbols = dict() # type: Dict[str, int]
96  self.local_dims = dict() # type: Dict[str, int]
97  self.allow_new_symbols = allow_new_symbols
98  self.allow_new_dims = allow_new_dims
99 
100  def get_dim(self, dimname: str) -> int:
101  """Gets the dim position given a name."""
102  pos = self.all_dims.get(dimname)
103  if pos is None:
104  if not self.allow_new_dims:
105  raise ValueError(
106  f"New dimensions not allowed in the current affine expression: "
107  f"Requested '{dimname}', Availble: {self.all_dims}")
108  pos = len(self.all_dims)
109  self.all_dims[dimname] = pos
110  self.local_dims[dimname] = pos
111  return pos
112 
113  def get_symbol(self, symname: str) -> int:
114  """Geta a symbol position given a name."""
115  pos = self.all_symbols.get(symname)
116  if pos is None:
117  if not self.allow_new_symbols:
118  raise ValueError(
119  f"New symbols not allowed in the current affine expression: "
120  f"Requested '{symname}', Availble: {self.all_symbols}")
121  pos = len(self.all_symbols)
122  self.all_symbols[symname] = pos
123  self.local_symbols[symname] = pos
124  return pos
