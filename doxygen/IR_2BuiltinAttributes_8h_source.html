37 /// A base attribute that represents a reference to a static shaped tensor or
38 /// vector constant.
39 class ElementsAttr : public Attribute {
40 public:
41  using Attribute::Attribute;
42  template <typename T> using iterator = detail::ElementsAttrIterator<T>;
43  template <typename T> using iterator_range = detail::ElementsAttrRange<T>;
44 
45  /// Return the type of this ElementsAttr, guaranteed to be a vector or tensor
46  /// with static shape.
47  ShapedType getType() const;
48 
49  /// Return the value at the given index. The index is expected to refer to a
50  /// valid element.
51  Attribute getValue(ArrayRef<uint64_t> index) const;
52 
53  /// Return the value of type 'T' at the given index, where 'T' corresponds to
54  /// an Attribute type.
55  template <typename T> T getValue(ArrayRef<uint64_t> index) const {
56  return getValue(index).template cast<T>();
57  }
58 
59  /// Return the elements of this attribute as a value of type 'T'. Note:
60  /// Aborts if the subclass is OpaqueElementsAttrs, these attrs do not support
61  /// iteration.
62  template <typename T> iterator_range<T> getValues() const;
63 
64  /// Return if the given 'index' refers to a valid element in this attribute.
65  bool isValidIndex(ArrayRef<uint64_t> index) const;
66 
67  /// Returns the number of elements held by this attribute.
68  int64_t getNumElements() const;
69 
70  /// Returns the number of elements held by this attribute.
71  int64_t size() const { return getNumElements(); }
72 
73  /// Returns if the number of elements held by this attribute is 0.
74  bool empty() const { return size() == 0; }
75 
76  /// Generates a new ElementsAttr by mapping each int value to a new
77  /// underlying APInt. The new values can represent either an integer or float.
78  /// This ElementsAttr should contain integers.
79  ElementsAttr mapValues(Type newElementType,
80  function_ref<APInt(const APInt &)> mapping) const;
81 
82  /// Generates a new ElementsAttr by mapping each float value to a new
83  /// underlying APInt. The new values can represent either an integer or float.
84  /// This ElementsAttr should contain floats.
85  ElementsAttr mapValues(Type newElementType,
86  function_ref<APInt(const APFloat &)> mapping) const;
87 
88  /// Method for support type inquiry through isa, cast and dyn_cast.
89  static bool classof(Attribute attr);
90 
91 protected:
92  /// Returns the 1 dimensional flattened row-major index from the given
93  /// multi-dimensional index.
94  uint64_t getFlattenedIndex(ArrayRef<uint64_t> index) const;
95 };

153  /// Type trait used to check if the given type T is a potentially valid C++

157  template <typename T> struct is_valid_cpp_fp_type {
158  /// The type is a valid floating point type if it is a builtin floating
159  /// point type, or is a potentially user defined floating point type. The
160  /// latter allows for supporting users that have custom types defined for
161  /// bfloat16/half/etc.
162  static constexpr bool value = llvm::is_one_of<T, float, double>::value ||
163  (std::numeric_limits<T>::is_specialized &&

170  /// Constructs a dense elements attribute from an array of element values.
171  /// Each element attribute value is expected to be an element of 'type'.
172  /// 'type' must be a vector or tensor with static shape. If the element of
173  /// `type` is non-integer/index/float it is assumed to be a string type.
174  static DenseElementsAttr get(ShapedType type, ArrayRef<Attribute> values);
175 
176  /// Constructs a dense integer elements attribute from an array of integer
177  /// or floating-point values. Each value is expected to be the same bitwidth
178  /// of the element type of 'type'. 'type' must be a vector or tensor with
179  /// static shape.
180  template <typename T, typename = typename std::enable_if<
181  std::numeric_limits<T>::is_integer ||
182  is_valid_cpp_fp_type<T>::value>::type>
183  static DenseElementsAttr get(const ShapedType &type, ArrayRef<T> values) {
184  const char *data = reinterpret_cast<const char *>(values.data());
185  return getRawIntOrFloat(
186  type, ArrayRef<char>(data, values.size() * sizeof(T)), sizeof(T),
187  std::numeric_limits<T>::is_integer, std::numeric_limits<T>::is_signed);
188  }
189 
190  /// Constructs a dense integer elements attribute from a single element.
191  template <typename T, typename = typename std::enable_if<
192  std::numeric_limits<T>::is_integer ||
193  is_valid_cpp_fp_type<T>::value ||
194  detail::is_complex_t<T>::value>::type>
195  static DenseElementsAttr get(const ShapedType &type, T value) {
196  return get(type, llvm::makeArrayRef(value));
197  }

199  /// Constructs a dense complex elements attribute from an array of complex
200  /// values. Each value is expected to be the same bitwidth of the element type
201  /// of 'type'. 'type' must be a vector or tensor with static shape.
202  template <typename T, typename ElementT = typename T::value_type,
203  typename = typename std::enable_if<
204  detail::is_complex_t<T>::value &&

211  std::numeric_limits<ElementT>::is_signed);

777 template <typename T>
778 auto SparseElementsAttr::getValues() const
779  -> llvm::iterator_range<iterator<T>> {
780  auto zeroValue = getZeroValue<T>();
781  auto valueIt = getValues().getValues<T>().begin();
782  const std::vector<ptrdiff_t> flatSparseIndices(getFlattenedSparseIndices());
783  // TODO: Move-capture flatSparseIndices when c++14 is available.
784  std::function<T(ptrdiff_t)> mapFn = [=](ptrdiff_t index) {
785  // Try to map the current index to one of the sparse indices.
786  for (unsigned i = 0, e = flatSparseIndices.size(); i != e; ++i)
787  if (flatSparseIndices[i] == index)
788  return *std::next(valueIt, i);
789  // Otherwise, return the zero value.
790  return zeroValue;
791  };
792  return llvm::map_range(llvm::seq<ptrdiff_t>(0, getNumElements()), mapFn);
793 }

796 /// This class represents a general iterator over the values of an ElementsAttr.
797 /// It supports all subclasses aside from OpaqueElementsAttr.
798 template <typename T>
799 class ElementsAttrIterator
800  : public llvm::iterator_facade_base<ElementsAttrIterator<T>,
801  std::random_access_iterator_tag, T,
802  std::ptrdiff_t, T, T> {
803  // NOTE: We use a dummy enable_if here because MSVC cannot use 'decltype'
804  // inside of a conversion operator.
805  using DenseIteratorT = typename std::enable_if<
806  true,
807  decltype(std::declval<DenseElementsAttr>().getValues<T>().begin())>::type;
808  using SparseIteratorT = SparseElementsAttr::iterator<T>;
809 
810  /// A union containing the specific iterators for each derived attribute kind.
811  union Iterator {
812  Iterator(DenseIteratorT &&it) : denseIt(std::move(it)) {}
813  Iterator(SparseIteratorT &&it) : sparseIt(std::move(it)) {}
814  Iterator() {}
815  ~Iterator() {}
816 
817  operator const DenseIteratorT &() const { return denseIt; }
818  operator const SparseIteratorT &() const { return sparseIt; }
819  operator DenseIteratorT &() { return denseIt; }
820  operator SparseIteratorT &() { return sparseIt; }
821 
822  /// An instance of a dense elements iterator.
823  DenseIteratorT denseIt;
824  /// An instance of a sparse elements iterator.
825  SparseIteratorT sparseIt;
826  };
827 
828  /// Utility method to process a functor on each of the internal iterator
829  /// types.
830  template <typename RetT, template <typename> class ProcessFn,
831  typename... Args>
832  RetT process(Args &...args) const {
833  if (attr.isa<DenseElementsAttr>())
834  return ProcessFn<DenseIteratorT>()(args...);
835  if (attr.isa<SparseElementsAttr>())
836  return ProcessFn<SparseIteratorT>()(args...);
837  llvm_unreachable("unexpected attribute kind");
838  }
839 
840  /// Utility functors used to generically implement the iterators methods.
841  template <typename ItT> struct PlusAssign {
842  void operator()(ItT &it, ptrdiff_t offset) { it += offset; }
843  };
844  template <typename ItT> struct Minus {
845  ptrdiff_t operator()(const ItT &lhs, const ItT &rhs) { return lhs - rhs; }
846  };
847  template <typename ItT> struct MinusAssign {
848  void operator()(ItT &it, ptrdiff_t offset) { it -= offset; }
849  };
850  template <typename ItT> struct Dereference {
851  T operator()(ItT &it) { return *it; }
852  };
853  template <typename ItT> struct ConstructIter {
854  void operator()(ItT &dest, const ItT &it) { ::new (&dest) ItT(it); }
855  };
856  template <typename ItT> struct DestructIter {
857  void operator()(ItT &it) { it.~ItT(); }
858  };
859 
860 public:
861  ElementsAttrIterator(const ElementsAttrIterator<T> &rhs) : attr(rhs.attr) {
862  process<void, ConstructIter>(it, rhs.it);
863  }
864  ~ElementsAttrIterator() { process<void, DestructIter>(it); }
865 
866  /// Methods necessary to support random access iteration.
867  ptrdiff_t operator-(const ElementsAttrIterator<T> &rhs) const {
868  assert(attr == rhs.attr && "incompatible iterators");
869  return process<ptrdiff_t, Minus>(it, rhs.it);
870  }
871  bool operator==(const ElementsAttrIterator<T> &rhs) const {
872  return rhs.attr == attr && process<bool, std::equal_to>(it, rhs.it);
873  }
874  bool operator<(const ElementsAttrIterator<T> &rhs) const {
875  assert(attr == rhs.attr && "incompatible iterators");
876  return process<bool, std::less>(it, rhs.it);
877  }
878  ElementsAttrIterator<T> &operator+=(ptrdiff_t offset) {
879  process<void, PlusAssign>(it, offset);
880  return *this;
881  }
882  ElementsAttrIterator<T> &operator-=(ptrdiff_t offset) {
883  process<void, MinusAssign>(it, offset);
884  return *this;
885  }
886 
887  /// Dereference the iterator at the current index.
888  T operator*() { return process<T, Dereference>(it); }
889 
890 private:
891  template <typename IteratorT>
892  ElementsAttrIterator(Attribute attr, IteratorT &&it)
893  : attr(attr), it(std::forward<IteratorT>(it)) {}
894 
895  /// Allow accessing the constructor.
896  friend ElementsAttr;
897 
898  /// The parent elements attribute.
899  Attribute attr;
900 
901  /// A union containing the specific iterators for each derived kind.
902  Iterator it;
903 };
