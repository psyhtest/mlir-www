57 template <typename PatternType, typename OpType, typename OptionsType>
58 void enqueue(RewritePatternSet &patternList, OptionsType options,
59  StringRef opName, linalg::LinalgTransformationFilter m) {
60  if (!opName.empty())
61  patternList.add<PatternType>(opName, patternList.getContext(), options, m);
62  else
63  patternList.add<PatternType>(patternList.getContext(),
64  m.addOpFilter<OpType>(), options);
65 }

163 /// Codegen strategy controls how a Linalg op is progressively lowered.
164 /// The application uses a 3-level staged patterns strategy which allows
165 /// ordering transformations by using the Linalg `applyStagedPatterns`
166 /// function, where:
167 /// 1. The first stage consists of the successive `tile`, `promote` and
168 /// `vectorize` patterns, applied sequentially.
169 /// 2. The second stage consists of common local canonicalization patterns
170 /// that are applied eagerly after each stage-1 pattern.
171 /// 3. the third stage consists of more global transformation, also applied
172 /// eagerly, after all stage-2 patterns. Such more global transformations

180  transformationSequence.emplace_back(
181  std::make_unique<Tile<LinalgOpType>>(options, f));
182  return *this;
183  }
