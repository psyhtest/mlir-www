36 /// Entry point for unrolling declarative pattern rewrites.
37 /// `op` is unrolled to the `targetShape` as follows, for each of its operands:
38 /// 1. the unrolled type `unrolledVectorType` and number of unrolled instances
39 /// `numUnrolledInstances` are computed from the `targetShape`. For now it is
40 /// assumed the unrolling factors divide the vector sizes.
41 /// 2. a fakeFork cast op is inserted that takes the operand and returns
42 /// `numUnrolledInstances` results of type `unrolledVectorType`.
43 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
44 /// result of the fakeFork cast op.
45 /// 4. a fakeJoin cast op takes all these results and merges them into a
46 /// single aggregate vector result whose size matches the original
47 /// non-unrolled op operand types.
48 ///
49 /// Example:
50 ///
51 /// opA(operand0, operand1) // numUnrolledInstances = 3
52 ///
53 /// operand0 operand1
54 /// | |
55 /// fork fork
56 /// <----------gather all fork ops --------->
57 /// /|\ /|\
58 /// f00 f01 f02 f10 f11 f12
59 /// <---------- clone op 3 times --------->
60 /// opA0(f00, f10), opA1(f01, f11), opA2(f02, f12)
61 /// \ | /
62 /// <-------------------- join ------------------------->
63 ///
64 /// Other local patterns then kick in iteratively (including DCE) and compose
65 /// until all the fakeFork and fakeJoin ops are removed.
66 ///
67 /// This will be extended in the future to support more advanced use cases than
68 /// simple pointwise ops.
69 SmallVector<Value, 1> unrollSingleResultVectorOp(OpBuilder &builder,
70  Operation *op,
71  ArrayRef<int64_t> targetShape);

121 /// Pattern to apply `unrollSingleResultVectorOp` to a `targetShape`
122 /// declaratively.
123 struct UnrollVectorPattern : public RewritePattern {
124  using FilterConstraintType = std::function<LogicalResult(Operation *op)>;
125  UnrollVectorPattern(MLIRContext *context, UnrollVectorOptions options)
126  : RewritePattern(MatchAnyOpTypeTag(), /*benefit=*/1, context),
127  options(options) {}
128  LogicalResult matchAndRewrite(Operation *op,
129  PatternRewriter &rewriter) const override {
130  if (options.filterConstraint && failed(options.filterConstraint(op)))
131  return failure();
132  if (!options.nativeShape) {
133  return op->emitError("vector unrolling expects the native shape or native"
134  "shape call back function to be set");
135  }
136  auto unrollableVectorOp = dyn_cast<VectorUnrollOpInterface>(op);
137  if (!unrollableVectorOp)
138  return failure();
139  auto maybeUnrollShape = unrollableVectorOp.getShapeForUnroll();
140  if (!maybeUnrollShape)
141  return failure();
142  Optional<SmallVector<int64_t, 4>> targetShape = options.nativeShape(op);
143  if (!targetShape)
144  return op->emitError("failed to get target shape for vector unroll");
145  auto maybeShapeRatio = shapeRatio(*maybeUnrollShape, *targetShape);
146  if (!maybeShapeRatio ||
147  llvm::all_of(*maybeShapeRatio, [](int64_t v) { return v == 1; }))
148  return failure();
149  if (isa<TransferWriteOp>(op)) {
150  SmallVector<Value, 1> result;
151  if (failed(unrollTransferWriteOp(rewriter, op, *targetShape, result)))
152  return failure();
153  rewriter.replaceOp(op, result);
154  return success();
155  }
156  if (op->getNumResults() != 1)
157  return failure();
158  auto resultVector = unrollSingleResultVectorOp(rewriter, op, *targetShape);
159  if (resultVector.size() != 1)
160  return failure();
161  rewriter.replaceOp(op, resultVector.front());
162  return success();
163  }
164 
165 private:
166  UnrollVectorOptions options;
167 };

169 /// Split a vector.transfer operation into an in-bounds (i.e., no out-of-bounds
170 /// masking) fastpath and a slowpath.
171 /// If `ifOp` is not null and the result is `success, the `ifOp` points to the
172 /// newly created conditional upon function return.
173 /// To accomodate for the fact that the original vector.transfer indexing may be
174 /// arbitrary and the slow path indexes @[0...0] in the temporary buffer, the
175 /// scf.if op returns a view and values of type index.
176 /// At this time, only vector.transfer_read case is implemented.
177 ///
178 /// Example (a 2-D vector.transfer_read):
179 /// ```
180 /// %1 = vector.transfer_read %0[...], %pad : memref<A...>, vector<...>
181 /// ```
182 /// is transformed into:
183 /// ```
184 /// %1:3 = scf.if (%inBounds) {
185 /// // fastpath, direct cast
186 /// memref.cast %A: memref<A...> to compatibleMemRefType
187 /// scf.yield %view : compatibleMemRefType, index, index
188 /// } else {
189 /// // slowpath, not in-bounds vector.transfer or linalg.copy.
190 /// memref.cast %alloc: memref<B...> to compatibleMemRefType
191 /// scf.yield %4 : compatibleMemRefType, index, index
192 // }
193 /// %0 = vector.transfer_read %1#0[%1#1, %1#2] {in_bounds = [true ... true]}
194 /// ```
195 /// where `alloc` is a top of the function alloca'ed buffer of one vector.
196 ///
197 /// Preconditions:
198 /// 1. `xferOp.permutation_map()` must be a minor identity map
199 /// 2. the rank of the `xferOp.memref()` and the rank of the `xferOp.vector()`
200 /// must be equal. This will be relaxed in the future but requires
201 /// rank-reducing subviews.
