87 class TensorUse(TensorExpression):
88  """A used tensor represented by its (tensor_name, indices).
89 
90  Note that forming a comprehension via direct assignment is performed through
91  __setitem__ on the TensorDef level. However, performing a reduction with
92  compound ops (+=, *=, etc) is done by doing a:
93  TensorDef.__getitem__
94  TensorUse.__iadd__
95  TensorDef.__setitem__
96  """

104  return ScalarArg(self.tensor_def.tensor_name).expr()

115  def _compute_reduce_dims(self, rhs: TensorExpression) -> Set[DimDef]:
116  """For implicit reductions, computes default reduction dims.
117 
118  Assumes that the rhs is the expression being reduced and self is being
119  reduced into. Any indices referenced on the rhs and not in self are
120  considered reduction dims and will be ordered as encountered on the rhs.
121  """
122  rhs_dims = rhs._get_all_dim_defs()
123  lhs_dims = self._get_all_dim_defs()
124  return rhs_dims - lhs_dims

160  def __getitem__(self, dims) -> TensorUse:
161  assert self.owner, "TensorDef is not attached to an op"
162  state = AffineBuildState(global_state=self.owner._affine_state,
163  allow_new_symbols=False)
164  if not isinstance(dims, tuple):
165  dims = (dims,) # Handle single subscript case.
166  # Special case: (None) is a 0d-scalar use.
167  if dims == (None,):

172  if not isinstance(expr_def, AffineExprDef):
173  raise KeyError(
174  "A TensorDef can only be subscripted by a tuple of affine dims")
175  exprs.append(expr_def)
176  return TensorUse(self, exprs)

229 class Comprehension:
230  """Represents a single comprehension."""
231 
232  def __init__(self, *bindings: Tuple[TensorUse, TensorExpression]):
233  self.definitions = list() # List[TensorUse]
234  self.values = list() # List[TensorExpression]
235 
236  # Find the lhs to reduction rhs.
237  for assign, value in bindings:
238  if isinstance(value, ReduceApply):
239  if value.lhs:
240  raise ValueError(f"Reduction expression already assigns: {value}")
241  value.lhs = assign
242  self.definitions.append(assign)
243  self.values.append(value)
244 
245  @property
246  def all_reduction_dims(self) -> Set[Tuple[DimDef, ...]]:
247  """Gets the reduction dims for the comprehension or None."""
248  result = set()
249  for use in self.values:
250  if isinstance(use, ReduceApply):
251  result.add(use.reduce.reduce_dims)
252  else:
253  result.add(tuple())
254  return result
255 
256  def __repr__(self):
257  if len(self.definitions) > 1:
258  defs_repr = f"({', '.join(repr(d) for d in self.definitions)})"
259  values_repr = f"({', '.join(repr(v) for v in self.values)})"
260  else:
261  defs_repr = f"{repr(self.definitions[0])}"
262  values_repr = f"{repr(self.values[0])}"
263 
264  return f"{defs_repr} = {values_repr}"
