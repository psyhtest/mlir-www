60  """Constructs an `init_tensor` with either static or dynamic sizes."""
61  context = get_default_loc_context(loc)
62  operands = []
63  attributes = {}
64  # TODO: Refactor the InitTensorOp to take an element type attribute and
65  # then use normal result type inference, unifying the Python and C++ side
66  # with a standard mechanism (versus stashing that in builders).
67  if sizes and isinstance(sizes[0], Value):
68  # Dynamic sizes.
69  operands.extend(sizes)
70  static_size_ints = [-1] * len(sizes)
71  result_type = RankedTensorType.get(static_size_ints, element_type)
72  else:
73  # Static sizes.
74  result_type = RankedTensorType.get(sizes, element_type)
75  static_size_ints = sizes
76

79  [IntegerAttr.get(index_type, s) for s in static_size_ints],
80  context=context)

82  operands=operands,
83  attributes=attributes,
84  loc=loc,
85  ip=ip)
