52 // DmaStartOp starts a non-blocking DMA operation that transfers data from a
53 // source memref to a destination memref. The source and destination memref need
54 // not be of the same dimensionality, but need to have the same elemental type.
55 // The operands include the source and destination memref's each followed by its
56 // indices, size of the data transfer in terms of the number of elements (of the
57 // elemental type of the memref), a tag memref with its indices, and optionally
58 // at the end, a stride and a number_of_elements_per_stride arguments. The tag
59 // location is used by a DmaWaitOp to check for completion. The indices of the
60 // source memref, destination memref, and the tag memref have the same
61 // restrictions as any load/store. The optional stride arguments should be of
62 // 'index' type, and specify a stride for the slower memory space (memory space
63 // with a lower memory space id), transferring chunks of
64 // number_of_elements_per_stride every stride until %num_elements are
65 // transferred. Either both or no stride arguments should be specified.
66 //
67 // For example, a DmaStartOp operation that transfers 256 elements of a memref
68 // '%src' in memory space 0 at indices [%i, %j] to memref '%dst' in memory space
69 // 1 at indices [%k, %l], would be specified as follows:
70 //
71 // %num_elements = constant 256
72 // %idx = constant 0 : index
73 // %tag = alloc() : memref<1 x i32, (d0) -> (d0), 4>
74 // dma_start %src[%i, %j], %dst[%k, %l], %num_elements, %tag[%idx] :
75 // memref<40 x 128 x f32>, (d0) -> (d0), 0>,
76 // memref<2 x 1024 x f32>, (d0) -> (d0), 1>,
77 // memref<1 x i32>, (d0) -> (d0), 2>
78 //
79 // If %stride and %num_elt_per_stride are specified, the DMA is expected to
80 // transfer %num_elt_per_stride elements every %stride elements apart from
81 // memory space 0 until %num_elements are transferred.
82 //
83 // dma_start %src[%i, %j], %dst[%k, %l], %num_elements, %tag[%idx], %stride,
84 // %num_elt_per_stride :
85 //
86 // TODO: add additional operands to allow source and destination striding, and
87 // multiple stride levels.
88 // TODO: Consider replacing src/dst memref indices with view memrefs.

154  /// Returns true if this is a DMA from a faster memory space to a slower one.
155  bool isDestMemorySpaceFaster() {
156  return (getSrcMemorySpace() < getDstMemorySpace());
157  }
158 
159  /// Returns true if this is a DMA from a slower memory space to a faster one.
160  bool isSrcMemorySpaceFaster() {
161  // Assumes that a lower number is for a slower memory space.
162  return (getDstMemorySpace() < getSrcMemorySpace());
163  }

165  /// Given a DMA start operation, returns the operand position of either the
166  /// source or destination memref depending on the one that is at the higher
167  /// level of the memory hierarchy. Asserts failure if neither is true.
168  unsigned getFasterMemPos() {
169  assert(isSrcMemorySpaceFaster() || isDestMemorySpaceFaster());
