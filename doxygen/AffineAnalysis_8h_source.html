97 // DependenceComponent contains state about the direction of a dependence as an
98 // interval [lb, ub] for an AffineForOp.
99 // Distance vectors components are represented by the interval [lb, ub] with
100 // lb == ub.
101 // Direction vectors components are represented by the interval [lb, ub] with
102 // lb < ub. Note that ub/lb == None means unbounded.
103 struct DependenceComponent {
104  // The AffineForOp Operation associated with this dependence component.
105  Operation *op;
106  // The lower bound of the dependence distance.

109  Optional<int64_t> ub;

113 /// Checks whether two accesses to the same memref access the same element.
114 /// Each access is specified using the MemRefAccess structure, which contains
115 /// the operation, indices and memref associated with the access. Returns
116 /// 'NoDependence' if it can be determined conclusively that the accesses do not
117 /// access the same memref element. If 'allowRAR' is true, will consider
118 /// read-after-read dependences (typically used by applications trying to
119 /// optimize input reuse).
120 // TODO: Wrap 'dependenceConstraints' and 'dependenceComponents' into a single
121 // struct.
122 // TODO: Make 'dependenceConstraints' optional arg.
123 struct DependenceResult {
124  enum ResultEnum {
125  HasDependence, // A dependence exists between 'srcAccess' and 'dstAccess'.
126  NoDependence, // No dependence exists between 'srcAccess' and 'dstAccess'.
127  Failure, // Dependence check failed due to unsupported cases.
128  } value;
129  DependenceResult(ResultEnum v) : value(v) {}
130 };
