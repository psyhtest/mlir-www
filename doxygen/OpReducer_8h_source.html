25 class OpReducer {

27  virtual ~OpReducer() = default;
28  /// According to rangeToKeep, try to reduce the given module. We implicitly
29  /// number each interesting operation and rangeToKeep indicates that if an
30  /// operation's number falls into certain range, then we will not try to
31  /// reduce that operation.

40 /// Reducer is a helper class to remove potential uninteresting operations from
41 /// module.
42 template <typename OpType>
43 class Reducer : public OpReducer {
44 public:
45  ~Reducer() override = default;
46 
47  int getNumTargetOps(ModuleOp module) const override {
48  return std::distance(module.getOps<OpType>().begin(),
49  module.getOps<OpType>().end());
50  }
51 
52  void reduce(ModuleOp module,
53  ArrayRef<ReductionNode::Range> rangeToKeep) override {
54  std::vector<Operation *> opsToRemove;
55  size_t keepIndex = 0;
56 
57  for (auto op : enumerate(module.getOps<OpType>())) {
58  int index = op.index();
59  if (keepIndex < rangeToKeep.size() &&
60  index == rangeToKeep[keepIndex].second)
61  ++keepIndex;
62  if (keepIndex == rangeToKeep.size() ||
63  index < rangeToKeep[keepIndex].first)
64  opsToRemove.push_back(op.value());
65  }
66 
67  for (Operation *o : opsToRemove) {
68  o->dropAllUses();
69  o->erase();
70  }
71  }
72 };
