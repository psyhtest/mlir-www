60 /// Trait to check if T provides a `regionBuilder` method.
61 template <typename T, typename... Args>
62 using has_region_builder = decltype(T::regionBuilder);
63 template <typename T>
64 using detect_has_region_builder = llvm::is_detected<has_region_builder, T>;
65 
66 /// SFINAE helper for single C++ class without a `regionBuilder` method (e.g.
67 /// an OpInterface).
68 template <typename OpType, typename = std::enable_if_t<
69  !detect_has_region_builder<OpType>::value>>
70 void addNamedOpBuilderImpl(
71  llvm::StringMap<LinalgDialect::RegionBuilderFunType> &map) {
72  // Do nothing.
73 }
74 
75 template <typename OpType,
76  typename = std::enable_if_t<detect_has_region_builder<OpType>::value>,
77  typename = void>
78 void addNamedOpBuilderImpl(
79  llvm::StringMap<LinalgDialect::RegionBuilderFunType> &map) {
80  map.insert(std::make_pair(
81  OpType::getOperationName(),
82  static_cast<LinalgDialect::RegionBuilderFunType>(OpType::regionBuilder)));
83 }
84 
85 template <typename... OpTypes>
86 void addNamedOpBuilders(
87  llvm::StringMap<LinalgDialect::RegionBuilderFunType> &map) {
88  (void)std::initializer_list<int>{0,
89  (addNamedOpBuilderImpl<OpTypes>(map), 0)...};
90 }

112 Type mlir::linalg::LinalgDialect::parseType(DialectAsmParser &parser) const {
113  // Parse the main keyword for the type.
114  StringRef keyword;
115  if (parser.parseKeyword(&keyword))
116  return Type();
117  MLIRContext *context = getContext();
118 
119  // Handle 'range' types.
120  if (keyword == "range")
121  return RangeType::get(context);
122 
123  parser.emitError(parser.getNameLoc(), "unknown Linalg type: " + keyword);
124  return Type();
125 }
