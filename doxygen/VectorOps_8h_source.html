48 /// Collect a set of patterns to split transfer read/write ops.
49 ///
50 /// These patterns unrolls transfer read/write ops if the vector consumers/
51 /// producers are extract/insert slices op. Transfer ops can map to hardware
52 /// load/store functionalities, where the vector size matters for bandwith
53 /// considerations. So these patterns should be collected separately, instead
54 /// of being generic canonicalization patterns. Also one can let the
55 /// `ignoreFilter` to return true to fail matching for fine-grained control.
56 void populateSplitVectorTransferPatterns(
57  RewritePatternSet &patterns,
58  std::function<bool(Operation *)> ignoreFilter = nullptr);
59 
60 /// Collect a set of leading one dimension removal patterns.
61 ///
62 /// These patterns insert vector.shape_cast to remove leading one dimensions
63 /// to expose more canonical forms of read/write/insert/extract operations.
64 /// With them, there are more chances that we can cancel out extract-insert
65 /// pairs or forward write-read pairs.
66 void populateCastAwayVectorLeadingOneDimPatterns(RewritePatternSet &patterns);
67 
68 /// Collect a set of patterns that bubble up/down bitcast ops.
69 ///
70 /// These patterns move vector.bitcast ops to be before insert ops or after
71 /// extract ops where suitable. With them, bitcast will happen on smaller
72 /// vectors and there are more chances to share extract/insert ops.
73 void populateBubbleVectorBitCastOpPatterns(RewritePatternSet &patterns);
74 
75 /// Collect a set of vector slices transformation patterns:
76 /// ExtractSlicesOpLowering, InsertSlicesOpLowering
77 /// Useful for clients that want to express all vector "slices"
78 /// ops in terms of more elementary vector "slice" ops. If all
79 /// "produced" tuple values are "consumed" (the most common
80 /// use for "slices" ops), this lowering removes all tuple related
81 /// operations as well (through DCE and folding). If tuple values
82 /// "leak" coming in, however, some tuple related ops will remain.

111 /// Enum to control the lowering of `vector.contract` operations.
112 enum class VectorContractLowering {
113  /// Progressively lower to finer grained `vector.contract` and dot-products.
114  Dot = 0,
115  /// Lower to `vector.matrix_multiply`, maps 1-1 to LLVM matrix intrinsics.
116  Matmul = 1,
117  /// Lower to `vector.outerproduct`.
118  OuterProduct = 2,
119 };
120 /// Enum to control the lowering of `vector.transpose` operations.
121 enum class VectorTransposeLowering {
122  /// Lower transpose into element-wise extract and inserts.
123  EltWise = 0,
124  /// Lower 2-D transpose to `vector.flat_transpose`, maps 1-1 to LLVM matrix
125  /// intrinsics.
126  Flat = 1,
127 };
128 /// Enum to control the splitting of `vector.transfer` operations into
129 /// in-bounds and out-of-bounds variants.
130 enum class VectorTransferSplit {
131  /// Do not split vector transfer operations.
132  None = 0,
133  /// Split using in-bounds + out-of-bounds vector.transfer operations.
134  VectorTransfer = 1,
135  /// Split using an in-bounds vector.transfer + linalg.fill + linalg.copy
136  /// operations.
137  LinalgCopy = 2,
138  /// Do not split vector transfer operation but instead mark it as "in-bounds".
139  ForceInBounds = 3
140 };
