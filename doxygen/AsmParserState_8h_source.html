25 /// This class represents state from a parsed MLIR textual format string. It is
26 /// useful for building additional analysis and language utilities on top of
27 /// textual MLIR. This should generally not be used for traditional compilation.
28 class AsmParserState {
29 public:
30  /// This class represents a definition within the source manager, containing
31  /// it's defining location and locations of any uses. SMDefinitions are only
32  /// provided for entities that have uses within an input file, e.g. SSA
33  /// values, Blocks, and Symbols.
34  struct SMDefinition {
35  SMDefinition() = default;
36  SMDefinition(llvm::SMRange loc) : loc(loc) {}
37 
38  /// The source location of the definition.
39  llvm::SMRange loc;
40  /// The source location of all uses of the definition.
41  SmallVector<llvm::SMRange> uses;
42  };
43 
44  /// This class represents the information for an operation definition within
45  /// an input file.
46  struct OperationDefinition {
47  struct ResultGroupDefinition {
48  /// The result number that starts this group.
49  unsigned startIndex;
50  /// The source definition of the result group.
51  SMDefinition definition;
52  };
53 
54  OperationDefinition(Operation *op, llvm::SMRange loc) : op(op), loc(loc) {}
55 
56  /// The operation representing this definition.
57  Operation *op;
58 
59  /// The source location for the operation, i.e. the location of its name.
60  llvm::SMRange loc;
61 
62  /// Source definitions for any result groups of this operation.
63  SmallVector<std::pair<unsigned, SMDefinition>> resultGroups;
64  };
65 
66  /// This class represents the information for a block definition within the
67  /// input file.
68  struct BlockDefinition {
69  BlockDefinition(Block *block, llvm::SMRange loc = {})
70  : block(block), definition(loc) {}
71 
72  /// The block representing this definition.
73  Block *block;
74 
75  /// The source location for the block, i.e. the location of its name, and
76  /// any uses it has.
77  SMDefinition definition;
78 
79  /// Source definitions for any arguments of this block.
80  SmallVector<SMDefinition> arguments;
81  };

84  ~AsmParserState();
