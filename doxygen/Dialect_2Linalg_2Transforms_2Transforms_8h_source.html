136 /// Fuse a sequence of linalg operations (`ops`) using tile-and-fuse. This
137 /// proceeds as follows:
138 /// - Find outer parallel loops in these ops that can be fused.
139 /// - Tile fusable outer parallel loops of the last operation in the sequence.
140 /// - Fuse the remaining operations with the tiled operation
141 ///
142 /// For example, consider the sequence of matmul below
143 ///
144 /// linalg.matmul ins(%arg0, %arg1 : memref<256x32xf32>, memref<32x32xf32>)
145 /// outs(%arg2 : memref<256x32xf32>)
146 /// linalg.matmul ins(%arg2, %arg3 : memref<256x32xf32>, memref<32x32xf32>)
147 /// outs(%arg4 : memref<256x32xf32>)
148 ///
149 /// It is legal to fuse the RAW dependence (through %arg2) by only fusing the
150 /// matmuls row-wise. For example, the fused computation for the above is shown
151 /// below. The outer `scf.parallel` loop is the "fused" loop obtained by tiling
152 /// along the rows of the matrix. The entire rows of the first matmul operation
153 /// need to be computed before they can be used for the second matmul. The
154 /// second matmul is further tiled (similar to normal tiling).
155 ///
156 /// #map0 = affine_map<(d0, d1)[s0] -> (d0 * 32 + s0 + d1)>
157 /// #map1 = affine_map<(d0, d1) -> (d0 * 32 + d1)>
158 /// scf.parallel (%arg5) = (%c0) to (%c256) step (%c16) {
159 /// %0 = subview %arg2[%arg5, 0] [16, 32] [1, 1]
160 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
161 /// %1 = subview %arg4[%arg5, 0] [16, 32] [1, 1]
162 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
163 /// %2 = subview %arg0[%arg5, 0] [16, 32] [1, 1]
164 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
165 /// %3 = subview %arg1[0, 0] [32, 32] [1, 1]
166 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
167 /// %4 = subview %arg3[0, 0] [32, 32] [1, 1]
168 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
169 /// linalg.matmul
170 /// ins(%2, %3 : memref<16x32xf32, #map0>, memref<32x32xf32, #map1>)
171 /// outs(%0 : memref<16x32xf32, #map0>)
172 /// linalg.matmul
173 /// ins(%0, %4 : memref<16x4xf32, #map0>, memref<4x8xf32, #map0>)
174 /// outs(%1 : memref<16x8xf32, #map0>)
175 /// }
176 ///
177 /// `tilingOptions` are used to tile the corresponding operation in `ops` (the
178 /// size of the former should be same as size of the latter. Based on how
179 /// tile+fuse is implemented, the fused loops are generated based on the last
180 /// operation in the sequence. For example, the tile sizes for the fused loops
181 /// is obtained from `tilingOptions.back()`. The following tiling options are
182 /// handled differently in tile+fuse (compared to tile only)
183 /// - Interchange of the tiling loops is not supported right now.
184 /// - Only the fused loops are distributed.
185 struct TiledAndFusedLinalgOps {
186  /// Operation obtained by tiling the last operation in sequence of `ops`
187  /// passed to `tileAndFuseLinalgOps`.
188  LinalgOp op;
189  /// The dimension of the loops that are fused.
190  std::set<unsigned> fusedLoopDims;
191  /// The generated fused operations (created within the fused loops).
192  SmallVector<LinalgOp, 1> fusedProducers;
193  /// The fused loop generated.
194  SmallVector<Operation *, 4> fusedLoops;
195 };

993 ///
994 /// Conversion expects ConvOp to have dimensions marked in the *mask* as
995 /// false of size 1. This ensures that the ConvOp can be lowered to vector
996 /// contraction of dimensions marked in the *mask* as true.
997 ///
998 /// A good example for vectorization is ConvNHWCOp which is 2D Conv op
999 /// with channels as the last dimension. Let's vectorize last 3 dimensions.
1000 /// The initial op definition looks like this:
1001 /// ```

1003 /// (memref<1x3x3x3xf32>, memref<1x3x3x3xf32>, memref<?x?x?x?xf32>)

1006 /// %arg1 (kernel) which is written into first entry of %arg2 (output). This is
1007 /// the ConvOp this pass expects and converts into:
1008 /// ```
1009 /// #map0 = affine_map<(d0, d1, d2) -> (d0, d1, d2)>
1010 /// #map1 = affine_map<(d0, d1, d2) -> ()>
1011 /// .....
1012 /// %0 = vector.transfer_read %arg0[%c0, %c0, %c0, %c0], %c0_f32
1013 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>

1015 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>
1016 /// %2 = vector.contract {indexing_maps = [#map0, #map0, #map1],
1017 /// iterator_types = ["reduction", "reduction", "reduction"]} %0, %1,
1018 /// %c0_f32 : vector<3x3x3xf32>, vector<3x3x3xf32> into f32
1019 /// store %2, %arg2[%c0, %c0, %c0, %c0] : memref<?x?x?x?xf32>
