31 //
32 // Internal support for storing and reading sparse tensors.
33 //
34 // The following memory-resident sparse storage schemes are supported:
35 //
36 // (a) A coordinate scheme for temporarily storing and lexicographically
37 // sorting a sparse tensor by index.
38 //
39 // (b) A "one-size-fits-all" sparse storage scheme defined by per-rank
40 // sparse/dense annnotations to be used by generated MLIR code.
41 //
42 // The following external formats are supported:
43 //
44 // (1) Matrix Market Exchange (MME): *.mtx
45 // https://math.nist.gov/MatrixMarket/formats.html
46 //
47 // (2) Formidable Repository of Open Sparse Tensors and Tools (FROSTT): *.tns
48 // http://frostt.io/tensors/file-formats.html
49 //

54 /// A sparse tensor element in coordinate scheme (value and indices).
55 /// For example, a rank-1 vector element would look like
56 /// ({i}, a[i])
57 /// and a rank-5 tensor element like
58 /// ({i,j,k,l,m}, a[i,j,k,l,m])
59 struct Element {
60  Element(const std::vector<uint64_t> &ind, double val)
61  : indices(ind), value(val){};
62  std::vector<uint64_t> indices;
63  double value;
64 };
65 
66 /// A memory-resident sparse tensor in coordinate scheme (collection of
67 /// elements). This data structure is used to read a sparse tensor from
68 /// external file format into memory and sort the elements lexicographically
69 /// by indices before passing it back to the client (most packed storage
70 /// formats require the elements to appear in lexicographic index order).
71 struct SparseTensor {
72 public:
73  SparseTensor(const std::vector<uint64_t> &szs, uint64_t capacity)
74  : sizes(szs), pos(0) {
75  elements.reserve(capacity);
76  }
77  /// Adds element as indices and value.
78  void add(const std::vector<uint64_t> &ind, double val) {
79  assert(getRank() == ind.size());
80  for (int64_t r = 0, rank = getRank(); r < rank; r++)
81  assert(ind[r] < sizes[r]); // within bounds
82  elements.emplace_back(Element(ind, val));
83  }
84  /// Sorts elements lexicographically by index.
85  void sort() { std::sort(elements.begin(), elements.end(), lexOrder); }
86  /// Primitive one-time iteration.
87  const Element &next() { return elements[pos++]; }
88  /// Returns rank.
89  uint64_t getRank() const { return sizes.size(); }
90  /// Getter for sizes array.
91  const std::vector<uint64_t> &getSizes() const { return sizes; }
92  /// Getter for elements array.
93  const std::vector<Element> &getElements() const { return elements; }
94 
95 private:
96  /// Returns true if indices of e1 < indices of e2.
97  static bool lexOrder(const Element &e1, const Element &e2) {
98  assert(e1.indices.size() == e2.indices.size());
99  for (int64_t r = 0, rank = e1.indices.size(); r < rank; r++) {
100  if (e1.indices[r] == e2.indices[r])
101  continue;
102  return e1.indices[r] < e2.indices[r];
103  }
104  return false;
105  }
106  std::vector<uint64_t> sizes; // per-rank dimension sizes
107  std::vector<Element> elements;
108  uint64_t pos;
109 };

143 /// A memory-resident sparse tensor using a storage scheme based on per-rank
144 /// annotations on dense/sparse. This data structure provides a bufferized
145 /// form of an imaginary SparseTensorType, until such a type becomes a
146 /// first-class citizen of MLIR. In contrast to generating setup methods for
147 /// each differently annotated sparse tensor, this method provides a convenient
148 /// "one-size-fits-all" solution that simply takes an input tensor and
149 /// annotations to implement all required setup in a general manner.
150 template <typename P, typename I, typename V>
151 class SparseTensorStorage : public SparseTensorStorageBase {
152 public:
153  /// Constructs sparse tensor storage scheme following the given
154  /// per-rank dimension dense/sparse annotations.
155  SparseTensorStorage(SparseTensor *tensor, bool *sparsity)
156  : sizes(tensor->getSizes()), pointers(getRank()), indices(getRank()) {
157  // Provide hints on capacity.
158  // TODO: needs fine-tuning based on sparsity
159  uint64_t nnz = tensor->getElements().size();
160  values.reserve(nnz);
161  for (uint64_t d = 0, s = 1, rank = getRank(); d < rank; d++) {
162  s *= sizes[d];
163  if (sparsity[d]) {
164  pointers[d].reserve(s + 1);
165  indices[d].reserve(s);
166  s = 1;
167  }
168  }
169  // Then setup the tensor.
170  traverse(tensor, sparsity, 0, nnz, 0);
171  }
172 
173  virtual ~SparseTensorStorage() {}
174 
175  uint64_t getRank() const { return sizes.size(); }
176 
177  uint64_t getDimSize(uint64_t d) override { return sizes[d]; }
178 
179  // Partially specialize these three methods based on template types.
180  void getPointers(std::vector<P> **out, uint64_t d) override {
181  *out = &pointers[d];
182  }
183  void getIndices(std::vector<I> **out, uint64_t d) override {
184  *out = &indices[d];
185  }
186  void getValues(std::vector<V> **out) override { *out = &values; }
187 
188 private:
189  /// Initializes sparse tensor storage scheme from a memory-resident
190  /// representation of an external sparse tensor. This method prepares
191  /// the pointers and indices arrays under the given per-rank dimension
192  /// dense/sparse annotations.
193  void traverse(SparseTensor *tensor, bool *sparsity, uint64_t lo, uint64_t hi,
194  uint64_t d) {
195  const std::vector<Element> &elements = tensor->getElements();
196  // Once dimensions are exhausted, insert the numerical values.
197  if (d == getRank()) {
198  values.push_back(lo < hi ? elements[lo].value : 0.0);
199  return;
200  }
201  // Prepare a sparse pointer structure at this dimension.
202  if (sparsity[d] && pointers[d].empty())
203  pointers[d].push_back(0);
204  // Visit all elements in this interval.
205  uint64_t full = 0;
206  while (lo < hi) {
207  // Find segment in interval with same index elements in this dimension.
208  unsigned idx = elements[lo].indices[d];
209  unsigned seg = lo + 1;
210  while (seg < hi && elements[seg].indices[d] == idx)
211  seg++;
212  // Handle segment in interval for sparse or dense dimension.
213  if (sparsity[d]) {
214  indices[d].push_back(idx);
215  } else {
216  for (; full < idx; full++)
217  traverse(tensor, sparsity, 0, 0, d + 1); // pass empty
218  full++;
219  }
220  traverse(tensor, sparsity, lo, seg, d + 1);
221  // And move on to next segment in interval.
222  lo = seg;
223  }
224  // Finalize the sparse pointer structure at this dimension.
225  if (sparsity[d]) {
226  pointers[d].push_back(indices[d].size());
227  } else {
228  for (uint64_t sz = tensor->getSizes()[d]; full < sz; full++)
229  traverse(tensor, sparsity, 0, 0, d + 1); // pass empty
230  }
231  }
232 
233 private:
234  std::vector<uint64_t> sizes; // per-rank dimension sizes
235  std::vector<std::vector<P>> pointers;
236  std::vector<std::vector<I>> indices;
237  std::vector<V> values;
238 };

259 /// Read the MME header of a general sparse matrix of type real.
260 static void readMMEHeader(FILE *file, char *name, uint64_t *idata) {
261  char line[1025];
262  char header[64];
263  char object[64];
264  char format[64];
265  char field[64];
266  char symmetry[64];
267  // Read header line.
268  if (fscanf(file, "%63s %63s %63s %63s %63s\n", header, object, format, field,
269  symmetry) != 5) {
270  fprintf(stderr, "Corrupt header in %s\n", name);
271  exit(1);
272  }
273  // Make sure this is a general sparse matrix.
274  if (strcmp(toLower(header), "%%matrixmarket") ||
275  strcmp(toLower(object), "matrix") ||
276  strcmp(toLower(format), "coordinate") || strcmp(toLower(field), "real") ||
277  strcmp(toLower(symmetry), "general")) {
278  fprintf(stderr,
279  "Cannot find a general sparse matrix with type real in %s\n", name);
280  exit(1);
281  }
282  // Skip comments.
283  while (1) {
284  if (!fgets(line, 1025, file)) {
285  fprintf(stderr, "Cannot find data in %s\n", name);
286  exit(1);
287  }
288  if (line[0] != '%')
289  break;
290  }
291  // Next line contains M N NNZ.
292  idata[0] = 2; // rank
293  if (sscanf(line, "%" PRIu64 "%" PRIu64 "%" PRIu64 "\n", idata + 2, idata + 3,
294  idata + 1) != 3) {
295  fprintf(stderr, "Cannot find size in %s\n", name);
296  exit(1);
297  }
298 }

300 /// Read the "extended" FROSTT header. Although not part of the documented
301 /// format, we assume that the file starts with optional comments followed
302 /// by two lines that define the rank, the number of nonzeros, and the
303 /// dimensions sizes (one per rank) of the sparse tensor.
304 static void readExtFROSTTHeader(FILE *file, char *name, uint64_t *idata) {
305  char line[1025];
306  // Skip comments.
307  while (1) {
308  if (!fgets(line, 1025, file)) {
309  fprintf(stderr, "Cannot find data in %s\n", name);
310  exit(1);
311  }
312  if (line[0] != '#')
313  break;
314  }
315  // Next line contains RANK and NNZ.
316  if (sscanf(line, "%" PRIu64 "%" PRIu64 "\n", idata, idata + 1) != 2) {
317  fprintf(stderr, "Cannot find metadata in %s\n", name);
318  exit(1);
319  }
320  // Followed by a line with the dimension sizes (one per rank).
321  for (uint64_t r = 0; r < idata[0]; r++) {
322  if (fscanf(file, "%" PRIu64, idata + 2 + r) != 1) {
323  fprintf(stderr, "Cannot find dimension size %s\n", name);
324  exit(1);
325  }
326  }
327 }

332 //
333 // Public API of the sparse runtime support library that enables MLIR code
334 // to read a sparse tensor from an external format (MME for FROSTT).
335 //
336 // For example, a sparse matrix in MME can be read as follows.
337 //
338 // %tensor = call @openTensor(%fileName, %idata)
339 // : (!llvm.ptr<i8>, memref<?xindex>) -> (!llvm.ptr<i8>)
340 // %rank = load %idata[%c0] : memref<?xindex> # always 2 for MME
341 // %nnz = load %idata[%c1] : memref<?xindex>
342 // %m = load %idata[%c2] : memref<?xindex>
343 // %n = load %idata[%c3] : memref<?xindex>
344 // .. prepare reading in m x n sparse tensor A with nnz nonzero elements ..
345 // scf.for %k = %c0 to %nnz step %c1 {
346 // call @readTensorItem(%tensor, %idata, %ddata)
347 // : (!llvm.ptr<i8>, memref<?xindex>, memref<?xf64>) -> ()
348 // %i = load %idata[%c0] : memref<?xindex>
349 // %j = load %idata[%c1] : memref<?xindex>
350 // %d = load %ddata[%c0] : memref<?xf64>
351 // .. process next nonzero element A[i][j] = d
352 // where the elements appear in lexicographic order ..
353 // }
354 // call @closeTensor(%tensor) : (!llvm.ptr<i8>) -> ()
355 //
356 //
357 // Note that input parameters in the "MLIRized" version of a function mimic
358 // the data layout of a MemRef<?xT> (but cannot use a direct struct). The
359 // output parameter uses a direct struct.
360 //
361 //===----------------------------------------------------------------------===//

365 /// Reads in a sparse tensor with the given filename. The call yields a
366 /// pointer to an opaque memory-resident sparse tensor object that is only
367 /// understood by other methods in the sparse runtime support library. An
368 /// array parameter is used to pass the rank, the number of nonzero elements,
369 /// and the dimension sizes (one per rank).
370 void *openTensorC(char *filename, uint64_t *idata) {
371  // Open the file.
372  FILE *file = fopen(filename, "r");
373  if (!file) {
374  fprintf(stderr, "Cannot find %s\n", filename);
375  exit(1);
376  }
377  // Perform some file format dependent set up.
378  if (strstr(filename, ".mtx")) {
379  readMMEHeader(file, filename, idata);
380  } else if (strstr(filename, ".tns")) {
381  readExtFROSTTHeader(file, filename, idata);
382  } else {
383  fprintf(stderr, "Unknown format %s\n", filename);
384  exit(1);
385  }
386  // Prepare sparse tensor object with per-rank dimension sizes
387  // and the number of nonzeros as initial capacity.
388  uint64_t rank = idata[0];
389  uint64_t nnz = idata[1];
390  std::vector<uint64_t> indices(rank);
391  for (uint64_t r = 0; r < rank; r++)
392  indices[r] = idata[2 + r];
393  SparseTensor *tensor = new SparseTensor(indices, nnz);
394  // Read all nonzero elements.
395  for (uint64_t k = 0; k < nnz; k++) {
396  for (uint64_t r = 0; r < rank; r++) {
397  if (fscanf(file, "%" PRIu64, &indices[r]) != 1) {
398  fprintf(stderr, "Cannot find next index in %s\n", filename);
399  exit(1);
400  }
401  indices[r]--; // 0-based index
402  }
403  double value;
404  if (fscanf(file, "%lg\n", &value) != 1) {
405  fprintf(stderr, "Cannot find next value in %s\n", filename);
406  exit(1);
407  }
408  tensor->add(indices, value);
409  }
410  // Close the file and return sorted tensor.
411  fclose(file);
412  tensor->sort(); // sort lexicographically
413  return tensor;
414 }
