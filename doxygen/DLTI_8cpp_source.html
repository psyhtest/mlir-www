64 /// Parses an attribute with syntax:
65 /// attr ::= `#target.` `dl_entry` `<` (type | quoted-string) `,` attr `>`
66 DataLayoutEntryAttr DataLayoutEntryAttr::parse(DialectAsmParser &parser) {
67  if (failed(parser.parseLess()))
68  return {};
69 
70  Type type = nullptr;
71  StringRef identifier;
72  llvm::SMLoc idLoc = parser.getCurrentLocation();
73  OptionalParseResult parsedType = parser.parseOptionalType(type);
74  if (parsedType.hasValue() && failed(parsedType.getValue()))
75  return {};
76  if (!parsedType.hasValue()) {
77  OptionalParseResult parsedString = parser.parseOptionalString(&identifier);
78  if (!parsedString.hasValue() || failed(parsedString.getValue())) {
79  parser.emitError(idLoc) << "expected a type or a quoted string";
80  return {};
81  }
82  }
83 
84  Attribute value;
85  if (failed(parser.parseComma()) || failed(parser.parseAttribute(value)) ||
86  failed(parser.parseGreater()))
87  return {};
88 
89  return type ? get(type, value)
90  : get(parser.getBuilder().getIdentifier(identifier), value);
91 }

144 DataLayoutSpecAttr::verify(function_ref<InFlightDiagnostic()> emitError,
145  ArrayRef<DataLayoutEntryInterface> entries) {
146  DenseSet<Type> types;
147  DenseSet<Identifier> ids;
148  for (DataLayoutEntryInterface entry : entries) {
149  if (auto type = entry.getKey().dyn_cast<Type>()) {
150  if (!types.insert(type).second)

161 /// Given a list of old and a list of new entries, overwrites old entries with
162 /// new ones if they have matching keys, appends new entries to the old entry
163 /// list otherwise.
164 static void
165 overwriteDuplicateEntries(SmallVectorImpl<DataLayoutEntryInterface> &oldEntries,
166  ArrayRef<DataLayoutEntryInterface> newEntries) {
167  unsigned oldEntriesSize = oldEntries.size();
168  for (DataLayoutEntryInterface entry : newEntries) {
169  // We expect a small (dozens) number of entries, so it is practically
170  // cheaper to iterate over the list linearly rather than to create an
171  // auxiliary hashmap to avoid duplication. Also note that we never need to
172  // check for duplicate keys the values that were added from `newEntries`.
173  bool replaced = false;
174  for (unsigned i = 0; i < oldEntriesSize; ++i) {
175  if (oldEntries[i].getKey() == entry.getKey()) {
176  oldEntries[i] = entry;
177  replaced = true;
178  break;
179  }
180  }
181  if (!replaced)
182  oldEntries.push_back(entry);
183  }
184 }

186 /// Combines a data layout spec into the given lists of entries organized by
187 /// type class and identifier, overwriting them if necessary. Fails to combine
188 /// if the two entries with identical keys are not compatible.
189 static LogicalResult
190 combineOneSpec(DataLayoutSpecInterface spec,
191  DenseMap<TypeID, DataLayoutEntryList> &entriesForType,
192  DenseMap<Identifier, DataLayoutEntryInterface> &entriesForID) {
193  // A missing spec should be fine.
194  if (!spec)
195  return success();
196 
197  DenseMap<TypeID, DataLayoutEntryList> newEntriesForType;
198  DenseMap<Identifier, DataLayoutEntryInterface> newEntriesForID;
199  spec.bucketEntriesByType(newEntriesForType, newEntriesForID);
200 
201  // Try overwriting the old entries with the new ones.
202  for (const auto &kvp : newEntriesForType) {
203  if (!entriesForType.count(kvp.first)) {
204  entriesForType[kvp.first] = std::move(kvp.second);
205  continue;
206  }
207 
208  Type typeSample = kvp.second.front().getKey().get<Type>();
209  assert(&typeSample.getDialect() !=
210  typeSample.getContext()->getLoadedDialect<BuiltinDialect>() &&
211  "unexpected data layout entry for built-in type");
212 
213  auto interface = typeSample.cast<DataLayoutTypeInterface>();
214  if (!interface.areCompatible(entriesForType.lookup(kvp.first), kvp.second))
215  return failure();
216 
217  overwriteDuplicateEntries(entriesForType[kvp.first], kvp.second);
218  }
219 
220  for (const auto &kvp : newEntriesForID) {
221  Identifier id = kvp.second.getKey().get<Identifier>();
222  Dialect *dialect = id.getDialect();
223  if (!entriesForID.count(id)) {
224  entriesForID[id] = kvp.second;
225  continue;
226  }
227 
228  // Attempt to combine the enties using the dialect interface. If the
229  // dialect is not loaded for some reason, use the default combinator
230  // that conservatively accepts identical entries only.
231  entriesForID[id] =
232  dialect ? dialect->getRegisteredInterface<DataLayoutDialectInterface>()
233  ->combine(entriesForID[id], kvp.second)
234  : DataLayoutDialectInterface::defaultCombine(entriesForID[id],
235  kvp.second);
236  if (!entriesForID[id])
237  return failure();
238  }
239 
240  return success();
241 }

243 DataLayoutSpecAttr
244 DataLayoutSpecAttr::combineWith(ArrayRef<DataLayoutSpecInterface> specs) const {
245  // Only combine with attributes of the same kind.
246  // TODO: reconsider this when the need arises.
247  if (llvm::any_of(specs, [](DataLayoutSpecInterface spec) {
248  return !spec.isa<DataLayoutSpecAttr>();
249  }))
250  return {};
251 
252  // Combine all specs in order, with `this` being the last one.
253  DenseMap<TypeID, DataLayoutEntryList> entriesForType;
254  DenseMap<Identifier, DataLayoutEntryInterface> entriesForID;
255  for (DataLayoutSpecInterface spec : specs)
256  if (failed(combineOneSpec(spec, entriesForType, entriesForID)))
257  return nullptr;
258  if (failed(combineOneSpec(*this, entriesForType, entriesForID)))
259  return nullptr;
260 
261  // Rebuild the linear list of entries.
262  SmallVector<DataLayoutEntryInterface> entries;
263  llvm::append_range(entries, llvm::make_second_range(entriesForID));
264  for (const auto &kvp : entriesForType)
265  llvm::append_range(entries, kvp.getSecond());
266 
267  return DataLayoutSpecAttr::get(getContext(), entries);
268 }

274 /// Parses an attribute with syntax
275 /// attr ::= `#target.` `dl_spec` `<` attr-list? `>`
276 /// attr-list ::= attr
277 /// | attr `,` attr-list
278 DataLayoutSpecAttr DataLayoutSpecAttr::parse(DialectAsmParser &parser) {
279  if (failed(parser.parseLess()))
280  return {};
281 
282  // Empty spec.
283  if (succeeded(parser.parseOptionalGreater()))
284  return get(parser.getBuilder().getContext(), {});
285 
286  SmallVector<DataLayoutEntryInterface> entries;
287  do {
288  entries.emplace_back();
289  if (failed(parser.parseAttribute(entries.back())))
290  return {};
291  } while (succeeded(parser.parseOptionalComma()));
292 
293  if (failed(parser.parseGreater()))
294  return {};
295  return getChecked([&] { return parser.emitError(parser.getNameLoc()); },
296  parser.getBuilder().getContext(), entries);
297 }
