63 /// A Callback<T> is a void function that accepts Expected<T>. This is
64 /// accepted by functions that logically return T.

77  template <typename T>
78  static llvm::Expected<T> parse(const llvm::json::Value &raw,
79  StringRef payloadName, StringRef payloadKind) {
80  T result;
81  llvm::json::Path::Root root;
82  if (fromJSON(raw, result, root))
83  return std::move(result);
84 
85  // Dump the relevant parts of the broken message.
86  std::string context;
87  llvm::raw_string_ostream os(context);
88  root.printErrorContext(raw, os);
89 
90  // Report the error (e.g. to the client).
91  return llvm::make_error<LSPError>(
92  llvm::formatv("failed to decode {0} {1}: {2}", payloadName, payloadKind,
93  fmt_consume(root.getError())),
94  ErrorCode::InvalidParams);
95  }
96 
97  template <typename Param, typename Result, typename ThisT>
98  void method(llvm::StringLiteral method, ThisT *thisPtr,
99  void (ThisT::*handler)(const Param &, Callback<Result>)) {
100  methodHandlers[method] = [method, handler,
101  thisPtr](llvm::json::Value rawParams,
102  Callback<llvm::json::Value> reply) {
103  llvm::Expected<Param> param = parse<Param>(rawParams, method, "request");
104  if (!param)
105  return reply(param.takeError());
106  (thisPtr->*handler)(*param, std::move(reply));
107  };
108  }
109 
110  template <typename Param, typename ThisT>
111  void notification(llvm::StringLiteral method, ThisT *thisPtr,
112  void (ThisT::*handler)(const Param &)) {
113  notificationHandlers[method] = [method, handler,
114  thisPtr](llvm::json::Value rawParams) {
115  llvm::Expected<Param> param = parse<Param>(rawParams, method, "request");
116  if (!param)
117  return llvm::consumeError(param.takeError());
118  (thisPtr->*handler)(*param);
119  };
120  }
121 
122 private:
123  template <typename HandlerT>
124  using HandlerMap = llvm::StringMap<llvm::unique_function<HandlerT>>;
125 
126  HandlerMap<void(llvm::json::Value)> notificationHandlers;
127  HandlerMap<void(llvm::json::Value, Callback<llvm::json::Value>)>
128  methodHandlers;

130  JSONTransport &transport;
