62 /// This is a common base class between Vector, UnrankedTensor, RankedTensor,
63 /// and MemRef types because they share behavior and semantics around shape,
64 /// rank, and fixed element type. Any type with these semantics should inherit
65 /// from ShapedType.
66 class ShapedType : public Type {
67 public:
68  using Type::Type;
69 
70  // TODO: merge these two special values in a single one used everywhere.
71  // Unfortunately, uses of `-1` have crept deep into the codebase now and are
72  // hard to track.
73  static constexpr int64_t kDynamicSize = -1;
74  static constexpr int64_t kDynamicStrideOrOffset =
75  std::numeric_limits<int64_t>::min();
76 
77  /// Return clone of this type with new shape and element type.
78  ShapedType clone(ArrayRef<int64_t> shape, Type elementType);
79  ShapedType clone(ArrayRef<int64_t> shape);
80  ShapedType clone(Type elementType);
81 
82  /// Return the element type.
83  Type getElementType() const;
84 
85  /// If an element type is an integer or a float, return its width. Otherwise,
86  /// abort.
87  unsigned getElementTypeBitWidth() const;
88 
89  /// If it has static shape, return the number of elements. Otherwise, abort.
90  int64_t getNumElements() const;
91 
92  /// If this is a ranked type, return the rank. Otherwise, abort.
93  int64_t getRank() const;
94 
95  /// Whether or not this is a ranked type. Memrefs, vectors and ranked tensors
96  /// have a rank, while unranked tensors do not.
97  bool hasRank() const;
98 
99  /// If this is a ranked type, return the shape. Otherwise, abort.
100  ArrayRef<int64_t> getShape() const;
101 
102  /// If this is unranked type or any dimension has unknown size (<0), it
103  /// doesn't have static shape. If all dimensions have known size (>= 0), it
104  /// has static shape.
105  bool hasStaticShape() const;
106 
107  /// If this has a static shape and the shape is equal to `shape` return true.
108  bool hasStaticShape(ArrayRef<int64_t> shape) const;
109 
110  /// If this is a ranked type, return the number of dimensions with dynamic
111  /// size. Otherwise, abort.
112  int64_t getNumDynamicDims() const;
113 
114  /// If this is ranked type, return the size of the specified dimension.
115  /// Otherwise, abort.
116  int64_t getDimSize(unsigned idx) const;
117 
118  /// Returns true if this dimension has a dynamic size (for ranked types);
119  /// aborts for unranked types.
120  bool isDynamicDim(unsigned idx) const;
121 
122  /// Returns the position of the dynamic dimension relative to just the dynamic
123  /// dimensions, given its `index` within the shape.
124  unsigned getDynamicDimIndex(unsigned index) const;
125 
126  /// Get the total amount of bits occupied by a value of this type. This does
127  /// not take into account any memory layout or widening constraints, e.g. a
128  /// vector<3xi57> is reported to occupy 3x57=171 bit, even though in practice
129  /// it will likely be stored as in a 4xi64 vector register. Fail an assertion
130  /// if the size cannot be computed statically, i.e. if the type has a dynamic
131  /// shape or if its elemental type does not have a known bit width.
132  int64_t getSizeInBits() const;
133 
134  /// Methods for support type inquiry through isa, cast, and dyn_cast.
135  static bool classof(Type type);
136 
137  /// Whether the given dimension size indicates a dynamic dimension.
138  static constexpr bool isDynamic(int64_t dSize) {
139  return dSize == kDynamicSize;
140  }
141  static constexpr bool isDynamicStrideOrOffset(int64_t dStrideOrOffset) {
142  return dStrideOrOffset == kDynamicStrideOrOffset;
143  }
144 };

314 /// Returns the strides of the MemRef if the layout map is in strided form.
315 /// MemRefs with layout maps in strided form include:
316 /// 1. empty or identity layout map, in which case the stride information is
317 /// the canonical form computed from sizes;
318 /// 2. single affine map layout of the form `K + k0 * d0 + ... kn * dn`,
319 /// where K and ki's are constants or symbols.
320 ///
321 /// A stride specification is a list of integer values that are either static
322 /// or dynamic (encoded with getDynamicStrideOrOffset()). Strides encode the
323 /// distance in the number of elements between successive entries along a
324 /// particular dimension. For example, `memref<42x16xf32, (64 * d0 + d1)>`
325 /// specifies a view into a non-contiguous memory region of `42` by `16` `f32`
326 /// elements in which the distance between two consecutive elements along the
327 /// outer dimension is `1` and the distance between two consecutive elements
328 /// along the inner dimension is `64`.
329 ///
330 /// Returns whether a simple strided form can be extracted from the composition
331 /// of the layout map.
332 ///
333 /// The convention is that the strides for dimensions d0, .. dn appear in
334 /// order to make indexing intuitive into the result.
335 LogicalResult getStridesAndOffset(MemRefType t,
336  SmallVectorImpl<int64_t> &strides,
337  int64_t &offset);
338 LogicalResult getStridesAndOffset(MemRefType t,
339  SmallVectorImpl<AffineExpr> &strides,
340  AffineExpr &offset);
341 
342 /// Given a list of strides (in which MemRefType::getDynamicStrideOrOffset()
343 /// represents a dynamic value), return the single result AffineMap which
344 /// represents the linearized strided layout map. Dimensions correspond to the
345 /// offset followed by the strides in order. Symbols are inserted for each
346 /// dynamic dimension in order. A stride cannot take value `0`.
347 ///
348 /// Examples:
349 /// =========
350 ///
351 /// 1. For offset: 0 strides: ?, ?, 1 return
352 /// (i, j, k)[M, N]->(M * i + N * j + k)
353 ///
354 /// 2. For offset: 3 strides: 32, ?, 16 return
355 /// (i, j, k)[M]->(3 + 32 * i + M * j + 16 * k)
356 ///
357 /// 3. For offset: ? strides: ?, ?, ? return
358 /// (i, j, k)[off, M, N, P]->(off + M * i + N * j + P * k)
359 AffineMap makeStridedLinearLayoutMap(ArrayRef<int64_t> strides, int64_t offset,
360  MLIRContext *context);
361 
362 /// Return a version of `t` with identity layout if it can be determined
363 /// statically that the layout is the canonical contiguous strided layout.
364 /// Otherwise pass `t`'s layout into `simplifyAffineMap` and return a copy of
365 /// `t` with simplified layout.
366 MemRefType canonicalizeStridedLayout(MemRefType t);
367 
368 /// Return a version of `t` with a layout that has all dynamic offset and
369 /// strides. This is used to erase the static layout.
370 MemRefType eraseStridedLayout(MemRefType t);
371 
372 /// Given MemRef `sizes` that are either static or dynamic, returns the
373 /// canonical "contiguous" strides AffineExpr. Strides are multiplicative and
374 /// once a dynamic dimension is encountered, all canonical strides become
375 /// dynamic and need to be encoded with a different symbol.
376 /// For canonical strides expressions, the offset is always 0 and and fastest
377 /// varying stride is always `1`.
378 ///
379 /// Examples:
380 /// - memref<3x4x5xf32> has canonical stride expression
381 /// `20*exprs[0] + 5*exprs[1] + exprs[2]`.
382 /// - memref<3x?x5xf32> has canonical stride expression
383 /// `s0*exprs[0] + 5*exprs[1] + exprs[2]`.
384 /// - memref<3x4x?xf32> has canonical stride expression
385 /// `s1*exprs[0] + s0*exprs[1] + exprs[2]`.
386 AffineExpr makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
387  ArrayRef<AffineExpr> exprs,
388  MLIRContext *context);
389 
390 /// Return the result of makeCanonicalStrudedLayoutExpr for the common case
391 /// where `exprs` is {d0, d1, .., d_(sizes.size()-1)}
392 AffineExpr makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
393  MLIRContext *context);
