29 def prepare_common_structured_op(op_config: LinalgStructuredOpConfig,
30  *ins: Value,
31  outs: Sequence[Value],
32  captures: Sequence[Value]):
33  all_arg_defs = op_config.ordered_tensor_args
34  in_arg_defs = [arg for arg in all_arg_defs if arg.usage == "input"]
35  out_arg_defs = [arg for arg in all_arg_defs if arg.usage == "output"]
36  capture_arg_defs = op_config.ordered_capture_args
37 
38  # Verify outs and captures are sequences.
39  if not isinstance(outs, Sequence):
40  raise ValueError(f"Expected named argument outs to have type Sequence "
41  f"but got {type(outs)}")
42  if not isinstance(captures, Sequence):
43  raise ValueError(f"Expected named argument captures to have type Sequence "
44  f"but got {type(outs)}")
45 
46  # Arity validation.
47  if len(ins) != len(in_arg_defs):
48  raise ValueError(f"Expected {len(in_arg_defs)} inputs but got "
49  f"{len(ins)} for {op_config}")
50  if outs and len(outs) != len(out_arg_defs):
51  raise ValueError(f"Expected {len(out_arg_defs)} outputs but got "
52  f"{len(outs)} for {op_config}")
53  if captures and len(captures) != len(capture_arg_defs):
54  raise ValueError(f"Expected {len(capture_arg_defs)} captures but got "
55  f"{len(captures)} for {op_config}")
56 
57  outs, out_types = _infer_structured_outs(op_config, in_arg_defs, ins,
58  out_arg_defs, outs)
59 
60  result_types = [t for t in out_types if isa(RankedTensorType, t)]
61 
62  # Initialize the type dictionary with the predefined types.
63  type_mapping = dict() # type: Dict[str, Type]
64  type_mapping["F32"] = F32Type.get()
65  type_mapping["F64"] = F64Type.get()
66  type_mapping["I32"] = IntegerType.get_signless(32)
67  type_mapping["I64"] = IntegerType.get_signless(64)
68 
69  # Extract type vars for input/output based types.
70  for arg_def, arg_element_type in zip(
71  in_arg_defs + out_arg_defs,
72  _get_shaped_element_types_from_values(*ins, *outs)):
73  _add_type_mapping(arg_def.tensor_def.type_var.name, arg_element_type,
74  type_mapping)
75 
76  # Extract type vars for captures and compute capture argument mapping.
77  capture_arg_mapping = dict() # type: Dict[str, Value]
78  for arg_def, capture_value in zip(capture_arg_defs, captures):
79  _add_type_mapping(arg_def.capture_def.type_var.name, capture_value.type,
80  type_mapping)
81  capture_arg_mapping[arg_def.capture_def.capture_name] = capture_value
82 
83  # Emit the generic op.
84  # TODO: Support emission of pure memref form.
85  indexing_maps_attr = ArrayAttr.get(
86  [AffineMapAttr.get(am)
87  # TODO: linalg verification does not currently allow symbols.
88  # Compress them for now.
89  for am in AffineMap.compress_unused_symbols(op_config.indexing_maps, Context.current)])
90  iterator_types_attr = ArrayAttr.get(
91  [StringAttr.get(s) for s in op_config.iterator_types])
92  # TODO: Add support for sparse operands once there is a stable interface.
93  sparse_attr = None
94 
95  return (all_arg_defs, in_arg_defs, out_arg_defs, outs, result_types,
96  type_mapping, capture_arg_mapping, indexing_maps_attr,
97  iterator_types_attr, sparse_attr)

100 def emit_generic_structured_op(op_config: LinalgStructuredOpConfig,
101  *ins: Value,
102  outs: Sequence[Value] = (),
103  captures: Sequence[Value] = ()):
104  all_arg_defs, in_arg_defs, out_arg_defs, outs, result_types, type_mapping, \
105  capture_arg_mapping, indexing_maps_attr, iterator_types_attr, sparse_attr = \
106  prepare_common_structured_op(op_config, *ins, outs = outs,
107  captures=captures)
108 
109  generic_op = linalg.GenericOp(
110  result_tensors=result_types,
111  inputs=ins,
112  outputs=outs,
113  indexing_maps=indexing_maps_attr,
114  iterator_types=iterator_types_attr,
115  doc=None, # TODO: Make optional.
116  library_call=None, # TODO: Make optional.
117  sparse=sparse_attr) # TODO: Make optional.
118 
119  # Construct the body.
120  block_arg_names = _get_tensor_def_names(*in_arg_defs, *out_arg_defs)
121  block_arg_types = _get_shaped_element_types_from_values(*ins, *outs)
122  block = generic_op.regions[0].blocks.append(*block_arg_types)
123  block_arg_mapping = dict(zip(block_arg_names, block.arguments))
124  with InsertionPoint(block):
125  body_builder = _BodyBuilder(type_mapping, block_arg_mapping,
126  capture_arg_mapping)
127  for assignment in op_config.assignments:
128  body_builder.assign(assignment)
129  body_builder.yield_outputs(*_get_tensor_def_names(*out_arg_defs))
130 
131  if len(result_types) == 1:
132  return generic_op.result
133  else:
134  return generic_op.results

138  op_name: str,
139  op_class_name: str,

146  captures = captures)
147 
148  # If we get here, there must exist a builtin class `op_class_name`.
149  ctx = Context.current
150  fully_qualified_name = 'linalg.' + op_name

163  # iterator_types are hardcoded in C++ both in the yaml and non-yaml path.
164 
165  if len(result_types) == 1:
166  return named_op.result
167  else:
168  return named_op.results

184  if assignment.arg in self.yield_mapping:

190  def expression(self, expr: ScalarExpression) -> Value:
191  if expr.scalar_arg:
192  try:
193  return self.block_arg_mapping[expr.scalar_arg.arg]
194  except KeyError:
195  raise ValueError(f"Argument {expr.scalar_arg.arg} is not bound for "
196  f"this structured op.")
197  elif expr.scalar_capture:
198  try:
199  return self.capture_arg_mapping[expr.scalar_capture.capture]
200  except KeyError:
201  raise ValueError(f"Capture {expr.scalar_capture.capture} is not bound for "
202  f"this structured op.")
203  elif expr.scalar_const:
204  return self.constant(expr.scalar_const.type_var.name, expr.scalar_const.value)
205  elif expr.scalar_index:
206  return self.index(expr.scalar_index.dim)
207  elif expr.scalar_apply:
208  try:
209  fn = getattr(self, f"_eval_{expr.scalar_apply.fn_name}")
210  except AttributeError:
211  raise ValueError(
212  f"Function '{expr.scalar_apply.fn_name}' is not a known "
213  "scalar body function")
214  operand_values = [
215  self.expression(operand) for operand in expr.scalar_apply.operands
216  ]
217  return fn(*operand_values)
218  elif expr.symbolic_cast:
219  operand_value = self.expression(expr.symbolic_cast.operand)
220  return self.cast(expr.symbolic_cast.to_type.name, operand_value)
221  raise NotImplementedError(f"Unimplemented scalar body expression: {expr}")

242  def cast(self, type_var_name: str, operand: Value) -> Value:
243  try:
244  to_type = self.type_mapping[type_var_name]
245  except KeyError:
246  raise ValueError(f"Unbound type variable '{type_var_name}' ("
247  f"expected one of {self.type_mappings.keys()}")
248  if operand.type == to_type:
249  return operand
250  if _is_integer_type(to_type):
251  return self._cast_to_integer(to_type, operand)
252  elif _is_floating_point_type(to_type):
253  return self._cast_to_floating_point(to_type, operand)
254  def _cast_to_integer(self, to_type: Type, operand: Value) -> Value:
255  to_width = IntegerType(to_type).width
256  operand_type = operand.type
257  if _is_floating_point_type(operand_type):
258  return std.FPToSIOp(to_type, operand).result
259  if _is_index_type(operand_type):
260  return std.IndexCastOp(to_type, operand).result
261  # Assume integer.
262  from_width = IntegerType(operand_type).width
263  if to_width > from_width:
264  return std.SignExtendIOp(to_type, operand).result
265  elif to_width < from_width:
266  return std.TruncateIOp(to_type, operand).result
267  raise ValueError(f"Unable to cast body expression from {operand_type} to "
268  f"{to_type}")
269 
270  def _cast_to_floating_point(self, to_type: Type, operand: Value) -> Value:
271  operand_type = operand.type
272  if _is_integer_type(operand_type):
273  return std.SIToFPOp(to_type, operand).result
274  # Assume FloatType.
275  to_width = _get_floating_point_width(to_type)
276  from_width = _get_floating_point_width(operand_type)
277  if to_width > from_width:
278  return std.FPExtOp(to_type, operand).result
279  elif to_width < from_width:
280  return std.FPTruncOp(to_type, operand).result
281  raise ValueError(f"Unable to cast body expression from {operand_type} to "
282  f"{to_type}")
