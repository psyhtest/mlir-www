11  def lookup(self, name):
12  """Lookup a function emitted with the `llvm.emit_c_interface`
13  attribute and returns a ctype callable.
14  Raise a RuntimeError if the function isn't found.
15  """

22  def invoke(self, name, *ctypes_args):
23  """Invoke a function with the list of ctypes arguments.
24  All arguments must be pointers.
25  Raise a RuntimeError if the function isn't found.
26  """

29  for argNum in range(len(ctypes_args)):
30  packed_args[argNum] = ctypes.cast(ctypes_args[argNum], ctypes.c_void_p)
31  func(packed_args)
