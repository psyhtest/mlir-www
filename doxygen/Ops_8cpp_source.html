37 /// Helper function to dispatch an OpFoldResult into either the `dynamicVec` if

63 /// Return true if ofr1 and ofr2 are the same integer constant attribute values
64 /// or the same SSA value.
65 /// Ignore integer bitwitdh and type mismatch that come from the fact there is
66 /// no IndexAttr and that IndexType have no bitwidth.
67 bool mlir::isEqualConstantIntOrValue(OpFoldResult op1, OpFoldResult op2) {
68  auto getConstantIntValue = [](OpFoldResult ofr) -> llvm::Optional<int64_t> {
69  Attribute attr = ofr.dyn_cast<Attribute>();
70  // Note: isa+cast-like pattern allows writing the condition below as 1 line.
71  if (!attr && ofr.get<Value>().getDefiningOp<ConstantOp>())
72  attr = ofr.get<Value>().getDefiningOp<ConstantOp>().getValue();
73  if (auto intAttr = attr.dyn_cast_or_null<IntegerAttr>())
74  return intAttr.getValue().getSExtValue();
75  return llvm::None;
76  };
77  auto cst1 = getConstantIntValue(op1), cst2 = getConstantIntValue(op2);
78  if (cst1 && cst2 && *cst1 == *cst2)
79  return true;
80  auto v1 = op1.dyn_cast<Value>(), v2 = op2.dyn_cast<Value>();
81  return v1 && v2 && v1 == v2;
82 }

441 /// Given a successor, try to collapse it to a new destination if it only
442 /// contains a passthrough unconditional branch. If the successor is
443 /// collapsable, `successor` and `successorOperands` are updated to reference
444 /// the new destination and values. `argStorage` is used as storage if operands
445 /// to the collapsed successor need to be remapped. It must outlive uses of
446 /// successorOperands.
447 static LogicalResult collapseBranch(Block *&successor,
448  ValueRange &successorOperands,
449  SmallVectorImpl<Value> &argStorage) {
450  // Check that the successor only contains a unconditional branch.
451  if (std::next(successor->begin()) != successor->end())
452  return failure();
453  // Check that the terminator is an unconditional branch.
454  BranchOp successorBranch = dyn_cast<BranchOp>(successor->getTerminator());
455  if (!successorBranch)
456  return failure();
457  // Check that the arguments are only used within the terminator.
458  for (BlockArgument arg : successor->getArguments()) {
459  for (Operation *user : arg.getUsers())
460  if (user != successorBranch)
461  return failure();
462  }
463  // Don't try to collapse branches to infinite loops.
464  Block *successorDest = successorBranch.getDest();
465  if (successorDest == successor)
466  return failure();
467 
468  // Update the operands to the successor. If the branch parent has no
469  // arguments, we can use the branch operands directly.
470  OperandRange operands = successorBranch.getOperands();
471  if (successor->args_empty()) {
472  successor = successorDest;
473  successorOperands = operands;
474  return success();
475  }
476 
477  // Otherwise, we need to remap any argument operands.
478  for (Value operand : operands) {
479  BlockArgument argOperand = operand.dyn_cast<BlockArgument>();
480  if (argOperand && argOperand.getOwner() == successor)
481  argStorage.push_back(successorOperands[argOperand.getArgNumber()]);
482  else
483  argStorage.push_back(operand);
484  }
485  successor = successorDest;
486  successorOperands = argStorage;
487  return success();
488 }

490 /// Simplify a branch to a block that has a single predecessor. This effectively
491 /// merges the two blocks.
492 static LogicalResult
493 simplifyBrToBlockWithSinglePred(BranchOp op, PatternRewriter &rewriter) {
494  // Check that the successor block has a single predecessor.
495  Block *succ = op.getDest();
496  Block *opParent = op->getBlock();
497  if (succ == opParent || !llvm::hasSingleElement(succ->getPredecessors()))
498  return failure();
499 
500  // Merge the successor into the current block and erase the branch.
501  rewriter.mergeBlocks(succ, opParent, op.getOperands());
502  rewriter.eraseOp(op);
503  return success();
504 }

506 /// br ^bb1
507 /// ^bb1
508 /// br ^bbN(...)
509 ///
510 /// -> br ^bbN(...)
511 ///
512 static LogicalResult simplifyPassThroughBr(BranchOp op,
513  PatternRewriter &rewriter) {
514  Block *dest = op.getDest();
515  ValueRange destOperands = op.getOperands();
516  SmallVector<Value, 4> destOperandStorage;
517 
518  // Try to collapse the successor if it points somewhere other than this
519  // block.
520  if (dest == op->getBlock() ||
521  failed(collapseBranch(dest, destOperands, destOperandStorage)))
522  return failure();
523 
524  // Create a new branch with the collapsed successor.
525  rewriter.replaceOpWithNewOp<BranchOp>(op, dest, destOperands);
526  return success();
527 }

553  // Check that the callee attribute was specified.
554  auto fnAttr = (*this)->getAttrOfType<FlatSymbolRefAttr>("callee");
555  if (!fnAttr)
556  return emitOpError("requires a 'callee' symbol reference attribute");

558  if (!fn)
559  return emitOpError() << "'" << fnAttr.getValue()
560  << "' does not reference a valid function";

563  auto fnType = fn.getType();

565  return emitOpError("incorrect number of operands for callee");

568  if (getOperand(i).getType() != fnType.getInput(i))
569  return emitOpError("operand type mismatch: expected operand type ")
570  << fnType.getInput(i) << ", but provided "
571  << getOperand(i).getType() << " for operand number " << i;
572 
573  if (fnType.getNumResults() != getNumResults())
574  return emitOpError("incorrect number of results for callee");
575 
576  for (unsigned i = 0, e = fnType.getNumResults(); i != e; ++i)
577  if (getResult(i).getType() != fnType.getResult(i))
578  return emitOpError("result type mismatch");
579 
580  return success();

712 /// Compute `lhs` `pred` `rhs`, where `pred` is one of the known floating point
713 /// comparison predicates.
714 bool mlir::applyCmpPredicate(CmpFPredicate predicate, const APFloat &lhs,
715  const APFloat &rhs) {
716  auto cmpResult = lhs.compare(rhs);
717  switch (predicate) {
718  case CmpFPredicate::AlwaysFalse:
719  return false;
720  case CmpFPredicate::OEQ:
721  return cmpResult == APFloat::cmpEqual;
722  case CmpFPredicate::OGT:
723  return cmpResult == APFloat::cmpGreaterThan;
724  case CmpFPredicate::OGE:
725  return cmpResult == APFloat::cmpGreaterThan ||
726  cmpResult == APFloat::cmpEqual;
727  case CmpFPredicate::OLT:
728  return cmpResult == APFloat::cmpLessThan;
729  case CmpFPredicate::OLE:
730  return cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
731  case CmpFPredicate::ONE:
732  return cmpResult != APFloat::cmpUnordered && cmpResult != APFloat::cmpEqual;
733  case CmpFPredicate::ORD:
734  return cmpResult != APFloat::cmpUnordered;
735  case CmpFPredicate::UEQ:
736  return cmpResult == APFloat::cmpUnordered || cmpResult == APFloat::cmpEqual;
737  case CmpFPredicate::UGT:
738  return cmpResult == APFloat::cmpUnordered ||
739  cmpResult == APFloat::cmpGreaterThan;
740  case CmpFPredicate::UGE:
741  return cmpResult == APFloat::cmpUnordered ||
742  cmpResult == APFloat::cmpGreaterThan ||
743  cmpResult == APFloat::cmpEqual;
744  case CmpFPredicate::ULT:
745  return cmpResult == APFloat::cmpUnordered ||
746  cmpResult == APFloat::cmpLessThan;
747  case CmpFPredicate::ULE:
748  return cmpResult == APFloat::cmpUnordered ||
749  cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
750  case CmpFPredicate::UNE:
751  return cmpResult != APFloat::cmpEqual;
752  case CmpFPredicate::UNO:
753  return cmpResult == APFloat::cmpUnordered;
754  case CmpFPredicate::AlwaysTrue:
755  return true;
756  }
757  llvm_unreachable("unknown comparison predicate");
758 }

806 /// cond_br %cond, ^bb1, ^bb2
807 /// ^bb1
808 /// br ^bbN(...)
809 /// ^bb2
810 /// br ^bbK(...)
811 ///
812 /// -> cond_br %cond, ^bbN(...), ^bbK(...)
813 ///
814 struct SimplifyPassThroughCondBranch : public OpRewritePattern<CondBranchOp> {
815  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
816 
817  LogicalResult matchAndRewrite(CondBranchOp condbr,
818  PatternRewriter &rewriter) const override {
819  Block *trueDest = condbr.trueDest(), *falseDest = condbr.falseDest();
820  ValueRange trueDestOperands = condbr.getTrueOperands();
821  ValueRange falseDestOperands = condbr.getFalseOperands();
822  SmallVector<Value, 4> trueDestOperandStorage, falseDestOperandStorage;
823 
824  // Try to collapse one of the current successors.
825  LogicalResult collapsedTrue =
826  collapseBranch(trueDest, trueDestOperands, trueDestOperandStorage);
827  LogicalResult collapsedFalse =
828  collapseBranch(falseDest, falseDestOperands, falseDestOperandStorage);
829  if (failed(collapsedTrue) && failed(collapsedFalse))
830  return failure();
831 
832  // Create a new branch with the collapsed successors.
833  rewriter.replaceOpWithNewOp<CondBranchOp>(condbr, condbr.getCondition(),
834  trueDest, trueDestOperands,
835  falseDest, falseDestOperands);
836  return success();
837  }
838 };

840 /// cond_br %cond, ^bb1(A, ..., N), ^bb1(A, ..., N)
841 /// -> br ^bb1(A, ..., N)
842 ///
843 /// cond_br %cond, ^bb1(A), ^bb1(B)
844 /// -> %select = select %cond, A, B
845 /// br ^bb1(%select)
846 ///
847 struct SimplifyCondBranchIdenticalSuccessors
848  : public OpRewritePattern<CondBranchOp> {
849  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
850 
851  LogicalResult matchAndRewrite(CondBranchOp condbr,
852  PatternRewriter &rewriter) const override {
853  // Check that the true and false destinations are the same and have the same
854  // operands.
855  Block *trueDest = condbr.trueDest();
856  if (trueDest != condbr.falseDest())
857  return failure();
858 
859  // If all of the operands match, no selects need to be generated.
860  OperandRange trueOperands = condbr.getTrueOperands();
861  OperandRange falseOperands = condbr.getFalseOperands();
862  if (trueOperands == falseOperands) {
863  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, trueOperands);
864  return success();
865  }
866 
867  // Otherwise, if the current block is the only predecessor insert selects
868  // for any mismatched branch operands.
869  if (trueDest->getUniquePredecessor() != condbr->getBlock())
870  return failure();
871 
872  // Generate a select for any operands that differ between the two.
873  SmallVector<Value, 8> mergedOperands;
874  mergedOperands.reserve(trueOperands.size());
875  Value condition = condbr.getCondition();
876  for (auto it : llvm::zip(trueOperands, falseOperands)) {
877  if (std::get<0>(it) == std::get<1>(it))
878  mergedOperands.push_back(std::get<0>(it));
879  else
880  mergedOperands.push_back(rewriter.create<SelectOp>(
881  condbr.getLoc(), condition, std::get<0>(it), std::get<1>(it)));
882  }
883 
884  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, mergedOperands);
885  return success();
886  }
887 };

889 /// ...
890 /// cond_br %cond, ^bb1(...), ^bb2(...)
891 /// ...
892 /// ^bb1: // has single predecessor
893 /// ...
894 /// cond_br %cond, ^bb3(...), ^bb4(...)
895 ///
896 /// ->
897 ///
898 /// ...
899 /// cond_br %cond, ^bb1(...), ^bb2(...)
900 /// ...
901 /// ^bb1: // has single predecessor
902 /// ...
903 /// br ^bb3(...)
904 ///
905 struct SimplifyCondBranchFromCondBranchOnSameCondition
906  : public OpRewritePattern<CondBranchOp> {
907  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
908 
909  LogicalResult matchAndRewrite(CondBranchOp condbr,
910  PatternRewriter &rewriter) const override {
911  // Check that we have a single distinct predecessor.
912  Block *currentBlock = condbr->getBlock();
913  Block *predecessor = currentBlock->getSinglePredecessor();
914  if (!predecessor)
915  return failure();
916 
917  // Check that the predecessor terminates with a conditional branch to this
918  // block and that it branches on the same condition.
919  auto predBranch = dyn_cast<CondBranchOp>(predecessor->getTerminator());
920  if (!predBranch || condbr.getCondition() != predBranch.getCondition())
921  return failure();
922 
923  // Fold this branch to an unconditional branch.
924  if (currentBlock == predBranch.trueDest())
925  rewriter.replaceOpWithNewOp<BranchOp>(condbr, condbr.trueDest(),
926  condbr.trueDestOperands());
927  else
928  rewriter.replaceOpWithNewOp<BranchOp>(condbr, condbr.falseDest(),
929  condbr.falseDestOperands());
930  return success();
931  }
932 };

990 /// The constant op requires an attribute, and furthermore requires that it
991 /// matches the return type.
992 static LogicalResult verify(ConstantOp &op) {
993  auto value = op.getValue();
994  if (!value)
995  return op.emitOpError("requires a 'value' attribute");
996 
997  Type type = op.getType();
998  if (!value.getType().isa<NoneType>() && type != value.getType())
999  return op.emitOpError() << "requires attribute's type (" << value.getType()
1000  << ") to match op's return type (" << type << ")";
1001 
1002  if (auto intAttr = value.dyn_cast<IntegerAttr>()) {
1003  if (type.isa<IndexType>() || value.isa<BoolAttr>())
1004  return success();
1005  IntegerType intType = type.cast<IntegerType>();
1006  if (!intType.isSignless())
1007  return op.emitOpError("requires integer result types to be signless");
1008 
1009  // If the type has a known bitwidth we verify that the value can be
1010  // represented with the given bitwidth.
1011  unsigned bitwidth = intType.getWidth();
1012  APInt intVal = intAttr.getValue();
1013  if (!intVal.isSignedIntN(bitwidth) && !intVal.isIntN(bitwidth))
1014  return op.emitOpError("requires 'value' to be an integer within the "
1015  "range of the integer result type");
1016  return success();
1017  }
1018 
1019  if (type.isa<FloatType>()) {
1020  if (!value.isa<FloatAttr>())
1021  return op.emitOpError("requires 'value' to be a floating point constant");
1022  return success();
1023  }
1024 
1025  if (type.isa<ShapedType>()) {
1026  if (!value.isa<ElementsAttr>())
1027  return op.emitOpError("requires 'value' to be a shaped constant");
1028  return success();
1029  }
1030 
1031  if (type.isa<FunctionType>()) {
1032  auto fnAttr = value.dyn_cast<FlatSymbolRefAttr>();
1033  if (!fnAttr)
1034  return op.emitOpError("requires 'value' to be a function reference");
1035 
1036  // Try to find the referenced function.
1037  auto fn =
1038  op->getParentOfType<ModuleOp>().lookupSymbol<FuncOp>(fnAttr.getValue());
1039  if (!fn)
1040  return op.emitOpError()
1041  << "reference to undefined function '" << fnAttr.getValue() << "'";
1042 
1043  // Check that the referenced function has the correct type.
1044  if (fn.getType() != type)
1045  return op.emitOpError("reference to function with mismatched type");
1046 
1047  return success();
1048  }
1049 
1050  if (type.isa<NoneType>() && value.isa<UnitAttr>())
1051  return success();
1052 
1053  return op.emitOpError("unsupported 'value' attribute: ") << value;
1054 }

1482 OpFoldResult SignedFloorDivIOp::fold(ArrayRef<Attribute> operands) {
1483  assert(operands.size() == 2 && "binary operation takes two operands");
1484 
1485  // Don't fold if it would overflow or if it requires a division by zero.
1486  bool overflowOrDiv0 = false;
1487  auto result = constFoldBinaryOp<IntegerAttr>(operands, [&](APInt a, APInt b) {
1488  if (overflowOrDiv0 || !b) {
1489  overflowOrDiv0 = true;
1490  return a;
1491  }
1492  unsigned bits = a.getBitWidth();
1493  APInt zero = APInt::getNullValue(bits);
1494  if (a.sge(zero) && b.sgt(zero)) {
1495  // Both positive (or a is zero), return a / b.
1496  return a.sdiv_ov(b, overflowOrDiv0);
1497  } else if (a.sle(zero) && b.slt(zero)) {
1498  // Both negative (or a is zero), return -a / -b.
1499  APInt posA = zero.ssub_ov(a, overflowOrDiv0);
1500  APInt posB = zero.ssub_ov(b, overflowOrDiv0);
1501  return posA.sdiv_ov(posB, overflowOrDiv0);
1502  } else if (a.slt(zero) && b.sgt(zero)) {
1503  // A is negative, b is positive, return - ceil(-a, b).
1504  APInt posA = zero.ssub_ov(a, overflowOrDiv0);
1505  APInt ceil = signedCeilNonnegInputs(posA, b, overflowOrDiv0);
1506  return zero.ssub_ov(ceil, overflowOrDiv0);
1507  } else {
1508  // A is positive, b is negative, return - ceil(a, -b).
1509  APInt posB = zero.ssub_ov(b, overflowOrDiv0);
1510  APInt ceil = signedCeilNonnegInputs(a, posB, overflowOrDiv0);
1511  return zero.ssub_ov(ceil, overflowOrDiv0);
1512  }
1513  });
1514 
1515  // Fold out floor division by one. Assumes all tensors of all ones are
1516  // splats.
1517  if (auto rhs = operands[1].dyn_cast_or_null<IntegerAttr>()) {
1518  if (rhs.getValue() == 1)

1842 /// Checks if `original` Type type can be rank reduced to `reduced` type.
1843 /// This function is slight variant of `is subsequence` algorithm where
1844 /// not matching dimension must be 1.
1845 static SubTensorVerificationResult
1846 isRankReducedType(Type originalType, Type candidateReducedType,
1847  std::string *errMsg = nullptr) {
1848  if (originalType == candidateReducedType)
1849  return SubTensorVerificationResult::Success;
1850  if (!originalType.isa<RankedTensorType>())
1851  return SubTensorVerificationResult::Success;
1852  if (originalType.isa<RankedTensorType>() &&
1853  !candidateReducedType.isa<RankedTensorType>())
1854  return SubTensorVerificationResult::Success;
1855 
1856  ShapedType originalShapedType = originalType.cast<ShapedType>();
1857  ShapedType candidateReducedShapedType =
1858  candidateReducedType.cast<ShapedType>();
1859 
1860  // Rank and size logic is valid for all ShapedTypes.
1861  ArrayRef<int64_t> originalShape = originalShapedType.getShape();
1862  ArrayRef<int64_t> candidateReducedShape =
1863  candidateReducedShapedType.getShape();
1864  unsigned originalRank = originalShape.size(),
1865  candidateReducedRank = candidateReducedShape.size();
1866  if (candidateReducedRank > originalRank)
1867  return SubTensorVerificationResult::RankTooLarge;
1868 
1869  auto optionalUnusedDimsMask =
1870  computeRankReductionMask(originalShape, candidateReducedShape);
1871 
1872  // Sizes cannot be matched in case empty vector is returned.
1873  if (!optionalUnusedDimsMask.hasValue())
1874  return SubTensorVerificationResult::SizeMismatch;
1875 
1876  if (originalShapedType.getElementType() !=
1877  candidateReducedShapedType.getElementType())
1878  return SubTensorVerificationResult::ElemTypeMismatch;
1879 
1880  // We are done for the tensor case.
1881  if (originalType.isa<RankedTensorType>())
1882  return SubTensorVerificationResult::Success;
1883 
1884  return SubTensorVerificationResult::Success;
