4 """Models DAGs of scalar math expressions.
5 
6 Used for generating region bodies at the "math" level where they are still type
7 polymorphic. This is modeled to be polymorphic by attribute name for interop
8 with serialization schemes that are just plain-old-dicts.
9 
10 These classes are typically not user accessed and are created as a by-product
11 of interpreting a comprehension DSL and model the operations to perform in the
12 op body. The class hierarchy is laid out to map well to a form of YAML that
13 can be easily consumed from the C++ side, not necessarily for ergonomics.
14 """

111 class ScalarExpression(YAMLObject):
112  """An expression on scalar values.
113 
114  Can be one of:
115  - ScalarApplyFn
116  - ScalarArg
117  - ScalarCapture
118  - ScalarConst
119  - ScalarIndex
120  - ScalarSymbolicCast
121  """
122  yaml_tag = "!ScalarExpression"
123 
124  def __init__(self,
125  scalar_apply: Optional[ScalarApplyFn] = None,
126  scalar_arg: Optional[ScalarArg] = None,
127  scalar_capture: Optional[ScalarCapture] = None,
128  scalar_const: Optional[ScalarConst] = None,
129  scalar_index: Optional[ScalarIndex] = None,
130  symbolic_cast: Optional[ScalarSymbolicCast] = None):
131  if (bool(scalar_apply) + bool(scalar_arg) + bool(scalar_capture) +
132  bool(scalar_const) + bool(scalar_index) + bool(symbolic_cast)) != 1:
133  raise ValueError(
134  "One of 'scalar_apply', 'scalar_arg', 'scalar_capture', 'scalar_const', "
135  "'scalar_index', 'symbolic_cast' must be specified")
136  self.scalar_apply = scalar_apply
137  self.scalar_arg = scalar_arg
138  self.scalar_capture = scalar_capture
139  self.scalar_const = scalar_const
140  self.scalar_index = scalar_index
141  self.symbolic_cast = symbolic_cast
142 
143  def to_yaml_custom_dict(self):
144  if self.scalar_apply:
145  return dict(scalar_apply=dict(
146  fn_name=self.scalar_apply.fn_name,
147  operands=list(self.scalar_apply.operands),
148  ))
149  elif self.scalar_arg:
150  return dict(scalar_arg=self.scalar_arg.arg)
151  elif self.scalar_capture:
152  return dict(scalar_capture=self.scalar_capture.capture)
153  elif self.scalar_const:
154  return dict(scalar_const=dict(type_var=self.scalar_const.type_var.name,
155  attributes=[self.scalar_const.value]))
156  elif self.scalar_index:
157  return dict(scalar_index=self.scalar_index.dim)
158  elif self.symbolic_cast:
159  # Note that even though operands must be arity 1, we write it the
160  # same way as for apply because it allows handling code to be more
161  # generic vs having a special form.
162  return dict(symbolic_cast=dict(type_var=self.symbolic_cast.to_type.name,
163  operands=[self.symbolic_cast.operand]))
164  else:
165  raise ValueError(f"Unexpected ScalarExpression type: {self}")
