54 /// Defines how the host (editor) should sync document changes to the language
55 /// server.
56 enum class TextDocumentSyncKind {
57  /// Documents should not be synced at all.
58  None = 0,
59 
60  /// Documents are synced by always sending the full content of the document.

94 /// URI in "file" scheme for a file.
95 class URIForFile {
96 public:
97  URIForFile() = default;
98 
99  /// Try to build a URIForFile from the given URI string.
100  static llvm::Expected<URIForFile> fromURI(StringRef uri);
101 
102  /// Try to build a URIForFile from the given absolute file path.
103  static llvm::Expected<URIForFile> fromFile(StringRef absoluteFilepath);
104 
105  /// Returns the absolute path to the file.
106  StringRef file() const { return filePath; }
107 
108  /// Returns the original uri of the file.
109  StringRef uri() const { return uriStr; }
110 
111  explicit operator bool() const { return !filePath.empty(); }
112 
113  friend bool operator==(const URIForFile &lhs, const URIForFile &rhs) {
114  return lhs.filePath == rhs.filePath;
115  }
116  friend bool operator!=(const URIForFile &lhs, const URIForFile &rhs) {
117  return !(lhs == rhs);
118  }
119  friend bool operator<(const URIForFile &lhs, const URIForFile &rhs) {
120  return lhs.filePath < rhs.filePath;
121  }
122 
123 private:
124  explicit URIForFile(std::string &&filePath, std::string &&uriStr)
125  : filePath(std::move(filePath)), uriStr(uriStr) {}
126 
127  std::string filePath;
128  std::string uriStr;
129 };

198 struct Position {
199  /// Line position in a document (zero-based).
200  int line = 0;
201 
202  /// Character offset on a line in a document (zero-based).
203  int character = 0;

205  friend bool operator==(const Position &lhs, const Position &rhs) {
206  return std::tie(lhs.line, lhs.character) ==
207  std::tie(rhs.line, rhs.character);
208  }
209  friend bool operator!=(const Position &lhs, const Position &rhs) {
210  return !(lhs == rhs);

212  friend bool operator<(const Position &lhs, const Position &rhs) {
213  return std::tie(lhs.line, lhs.character) <

215  }
216  friend bool operator<=(const Position &lhs, const Position &rhs) {
217  return std::tie(lhs.line, lhs.character) <=

230 struct Range {
231  /// The range's start position.
232  Position start;
233 
234  /// The range's end position.
235  Position end;
236 
237  friend bool operator==(const Range &lhs, const Range &rhs) {
238  return std::tie(lhs.start, lhs.end) == std::tie(rhs.start, rhs.end);
239  }
240  friend bool operator!=(const Range &lhs, const Range &rhs) {
241  return !(lhs == rhs);
242  }
243  friend bool operator<(const Range &lhs, const Range &rhs) {
244  return std::tie(lhs.start, lhs.end) < std::tie(rhs.start, rhs.end);
245  }
246 
247  bool contains(Position pos) const { return start <= pos && pos < end; }
248  bool contains(Range range) const {
249  return start <= range.start && range.end <= end;
250  }
