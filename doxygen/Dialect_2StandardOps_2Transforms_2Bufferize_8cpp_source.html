64 struct StdBufferizePass : public StdBufferizeBase<StdBufferizePass> {
65  void runOnFunction() override {
66  auto *context = &getContext();
67  BufferizeTypeConverter typeConverter;
68  RewritePatternSet patterns(context);
69  ConversionTarget target(*context);
70 
71  target.addLegalDialect<memref::MemRefDialect>();
72  target.addLegalDialect<StandardOpsDialect>();
73  target.addLegalDialect<scf::SCFDialect>();
74 
75  populateStdBufferizePatterns(typeConverter, patterns);
76  // We only bufferize the case of tensor selected type and scalar condition,
77  // as that boils down to a select over memref descriptors (don't need to
78  // touch the data).
79  target.addDynamicallyLegalOp<SelectOp>([&](SelectOp op) {
80  return typeConverter.isLegal(op.getType()) ||
81  !op.condition().getType().isa<IntegerType>();
82  });
83  target.addDynamicallyLegalOp<memref::DimOp>(
84  [&](memref::DimOp op) { return typeConverter.isLegal(op); });
85  if (failed(
86  applyPartialConversion(getFunction(), target, std::move(patterns))))
87  signalPassFailure();
88  }
89 };
90 } // namespace
