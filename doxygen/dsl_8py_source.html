48  """Emits the corresponding op definition as IR.
49 
50  Most arguments are passed through to the underlying emitter. The following
51  are interpreted here:
52  emit_generic: Emits a generic form as appropriate (default True). If
53  False, a named form is emitted (which must have been built in to the
54  compiler).
55  """
56  op_configs = LinalgOpConfig.from_linalg_op_def(self.model,
57  context=ir.Context.current)
58 
59  if len(op_configs) != 1:
60  # TODO: Support composite ops.
61  raise NotImplementedError(
62  f"Emission of composite linalg ops not supported: {op_configs}")
63 
64  ctx = ir.Context.current
65  linalgDialect = ctx.get_dialect_descriptor("linalg")
66  fully_qualified_name = 'linalg.' + self.op_name
67  emit_generic = (emit_generic or not ctx.is_registered_operation(fully_qualified_name))
68 
69  op_config = op_configs[0]
70  if op_config.structured_op:
71  if emit_generic:

83 def linalg_structured_op(dsl_func=None,
84  *,
85  op_name=None,
86  op_class_name=None) -> DefinedOpCallable:
87  if dsl_func is None:
88  # Curry the keyword args in for delayed application.
89  return functools.partial(tc_def_op,
90  op_name=op_name,
91  op_class_name=op_class_name)
92  # Determine default names by introspecting the function.
93  if op_name is None:
94  op_name = dsl_func.__name__
95  if op_class_name is None:
96  # Camel case it.
97  op_class_name = f"{''.join(x.title() for x in op_name.split('_'))}Op"
98 
99  tc_model = LinalgOpDef(name=op_name,
100  cpp_class_name=op_class_name,
101  doc=inspect.getdoc(dsl_func))
102 
103  # Extract arguments and TensorDefs from the signature.
104  dsl_func_args = list()
105  sig = inspect.signature(dsl_func)
106  for param_name, param in sig.parameters.items():
107  param_default = param.default
108  if isinstance(param_default, TensorDef):
109  tc_model.add_tensor(param_name, param_default)
110  elif isinstance(param_default, CaptureDef):
111  tc_model.add_capture(param_name, param_default)
112  else:
113  raise ValueError(f"@tc_def_op function parameters must be defaulted as "
114  f"TensorDef(...) or CaptureDef(...): Found {param_name}"
115  f": {param_default}")
116  dsl_func_args.append(param_default)
117 
118  # Invoke the DSL func to finish populating the model.
119  with bind_op_def(tc_model):
120  dsl_func(*dsl_func_args)
121 
122  # TODO: The returned callable should be an IR emitter but that is not
123  # upstreamed yet.
124  return DefinedOpCallable(op_name, tc_model)
