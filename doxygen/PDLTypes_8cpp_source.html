36 static Type parsePDLType(DialectAsmParser &parser) {
37  StringRef typeTag;
38  if (parser.parseKeyword(&typeTag))
39  return Type();
40  {
41  Type genType;
42  auto parseResult = generatedTypeParser(parser.getBuilder().getContext(),
43  parser, typeTag, genType);
44  if (parseResult.hasValue())
45  return genType;
46  }
47 
48  // FIXME: This ends up with a double error being emitted if `RangeType` also
49  // emits an error. We should rework the `generatedTypeParser` to better
50  // support when the keyword is valid but the individual type parser itself
51  // emits an error.
52  parser.emitError(parser.getNameLoc(), "invalid 'pdl' type: `")
53  << typeTag << "'";
54  return Type();
55 }

78 Type RangeType::parse(MLIRContext *context, DialectAsmParser &parser) {
79  if (parser.parseLess())
80  return Type();
81 
82  llvm::SMLoc elementLoc = parser.getCurrentLocation();
83  Type elementType = parsePDLType(parser);
84  if (!elementType || parser.parseGreater())
85  return Type();
86 
87  if (elementType.isa<RangeType>()) {
88  parser.emitError(elementLoc)
89  << "element of pdl.range cannot be another range, but got"
90  << elementType;
91  return Type();
92  }
93  return RangeType::get(elementType);
94 }
