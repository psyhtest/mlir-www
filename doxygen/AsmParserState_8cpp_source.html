14 /// Given a SMLoc corresponding to an identifier location, return a location
15 /// representing the full range of the identifier.
16 static llvm::SMRange convertIdLocToRange(llvm::SMLoc loc) {
17  if (!loc.isValid())
18  return llvm::SMRange();
19 
20  // Return if the given character is a valid identifier character.
21  auto isIdentifierChar = [](char c) {
22  return isalnum(c) || c == '$' || c == '.' || c == '_' || c == '-';
23  };
24 
25  const char *curPtr = loc.getPointer();
26  while (isIdentifierChar(*(++curPtr)))
27  continue;
28  return llvm::SMRange(loc, llvm::SMLoc::getFromPointer(curPtr));
29 }

70 void AsmParserState::addDefinition(
71  Operation *op, llvm::SMRange location,
72  ArrayRef<std::pair<unsigned, llvm::SMLoc>> resultGroups) {
73  std::unique_ptr<OperationDefinition> def =
74  std::make_unique<OperationDefinition>(op, location);
75  for (auto &resultGroup : resultGroups)
76  def->resultGroups.emplace_back(resultGroup.first,
77  convertIdLocToRange(resultGroup.second));
78 
79  impl->operationToIdx.try_emplace(op, impl->operations.size());
80  impl->operations.emplace_back(std::move(def));
81 }

83 void AsmParserState::addDefinition(Block *block, llvm::SMLoc location) {
84  auto it = impl->blocksToIdx.find(block);
85  if (it == impl->blocksToIdx.end()) {
86  impl->blocksToIdx.try_emplace(block, impl->blocks.size());
87  impl->blocks.emplace_back(std::make_unique<BlockDefinition>(
88  block, convertIdLocToRange(location)));
89  return;
90  }
91 
92  // If an entry already exists, this was a forward declaration that now has a
93  // proper definition.
94  impl->blocks[it->second]->definition.loc = convertIdLocToRange(location);
95 }

97 void AsmParserState::addDefinition(BlockArgument blockArg,
98  llvm::SMLoc location) {
99  auto it = impl->blocksToIdx.find(blockArg.getOwner());
100  assert(it != impl->blocksToIdx.end() &&
101  "expected owner block to have an entry");
102  BlockDefinition &def = *impl->blocks[it->second];
103  unsigned argIdx = blockArg.getArgNumber();
104 
105  if (def.arguments.size() <= argIdx)
106  def.arguments.resize(argIdx + 1);
107  def.arguments[argIdx] = SMDefinition(convertIdLocToRange(location));
108 }

110 void AsmParserState::addUses(Value value, ArrayRef<llvm::SMLoc> locations) {
111  // Handle the case where the value is an operation result.
112  if (OpResult result = value.dyn_cast<OpResult>()) {
113  // Check to see if a definition for the parent operation has been recorded.
114  // If one hasn't, we treat the provided value as a placeholder value that
115  // will be refined further later.
116  Operation *parentOp = result.getOwner();
117  auto existingIt = impl->operationToIdx.find(parentOp);
118  if (existingIt == impl->operationToIdx.end()) {
119  impl->placeholderValueUses[value].append(locations.begin(),
120  locations.end());
121  return;
122  }
123 
124  // If a definition does exist, locate the value's result group and add the
125  // use. The result groups are ordered by increasing start index, so we just
126  // need to find the last group that has a smaller/equal start index.
127  unsigned resultNo = result.getResultNumber();
128  OperationDefinition &def = *impl->operations[existingIt->second];
129  for (auto &resultGroup : llvm::reverse(def.resultGroups)) {
130  if (resultNo >= resultGroup.first) {
131  for (llvm::SMLoc loc : locations)
132  resultGroup.second.uses.push_back(convertIdLocToRange(loc));
133  return;
134  }
135  }
136  llvm_unreachable("expected valid result group for value use");
137  }
138 
139  // Otherwise, this is a block argument.
140  BlockArgument arg = value.cast<BlockArgument>();
141  auto existingIt = impl->blocksToIdx.find(arg.getOwner());
142  assert(existingIt != impl->blocksToIdx.end() &&
143  "expected valid block definition for block argument");
144  BlockDefinition &blockDef = *impl->blocks[existingIt->second];
145  SMDefinition &argDef = blockDef.arguments[arg.getArgNumber()];
146  for (llvm::SMLoc loc : locations)
147  argDef.uses.emplace_back(convertIdLocToRange(loc));
148 }

150 void AsmParserState::addUses(Block *block, ArrayRef<llvm::SMLoc> locations) {
151  auto it = impl->blocksToIdx.find(block);
152  if (it == impl->blocksToIdx.end()) {
153  it = impl->blocksToIdx.try_emplace(block, impl->blocks.size()).first;
154  impl->blocks.emplace_back(std::make_unique<BlockDefinition>(block));
155  }
156 
157  BlockDefinition &def = *impl->blocks[it->second];
158  for (llvm::SMLoc loc : locations)
159  def.definition.uses.push_back(convertIdLocToRange(loc));
160 }

162 void AsmParserState::refineDefinition(Value oldValue, Value newValue) {
163  auto it = impl->placeholderValueUses.find(oldValue);
164  assert(it != impl->placeholderValueUses.end() &&
165  "expected `oldValue` to be a placeholder");
166  addUses(newValue, it->second);
167  impl->placeholderValueUses.erase(oldValue);
168 }
