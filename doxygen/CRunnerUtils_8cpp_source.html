26 // Small runtime support "lib" for vector.print lowering.

44 /// Returns the number of seconds since Epoch 1970-01-01 00:00:00 +0000 (UTC).
45 extern "C" double rtclock() {
46 #ifndef _WIN32
47  struct timeval tp;
48  int stat = gettimeofday(&tp, NULL);
49  if (stat != 0)
50  fprintf(stderr, "Error returning time from gettimeofday: %d\n", stat);
51  return (tp.tv_sec + tp.tv_usec * 1.0e-6);
52 #else
53  fprintf(stderr, "Timing utility not implemented on Windows\n");
54  return 0.0;
55 #endif // _WIN32
56 }
