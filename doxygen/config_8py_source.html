115 class LinalgStructuredOpConfig(YAMLObject):
116  """Configuration for metadata sufficient to construct a linalg single
117  contraction named op."""
118 
119  yaml_tag = "!LinalgStructuredOpConfig"
120 
121  def __init__(self,
122  comprehension: Comprehension,
123  context: Optional[_ir.Context] = None):
124  self.context = context if context is not None else _ir.Context()
125  self.affine_state = AffineBuildState()
126  self.writes = list() # type: List[Tuple[TensorUse, TensorExpression]]
127  self.tensor_args = dict() # type: Dict[TensorDef, TensorDefConfig]
128  self.capture_args = dict() # type: Dict[CaptureDef, CaptureDefConfig]
129  self.uses = dict() # type: Dict[TensorUse, TensorUseConfig]
130 
131  # Compute the ordered set of writes and collect the tensor, capture, and
132  # index uses.
133  collected_uses = set()
134  collected_captures = set()
135  collected_indices = set()
136  for write_use, read_use in zip(comprehension.definitions,
137  comprehension.values):
138  self.writes.append((write_use, read_use))
139 
140  for write_use, read_use in self.writes:
141  collected_uses.add(write_use)
142  read_use.collect_uses(collected_uses)
143  read_use.collect_captures(collected_captures)
144  read_use.collect_indices(collected_indices)
145 
146  # Need to add all definitions before uses, so process twice.
147  for use in collected_uses:
148  self.add_tensor_arg(use.tensor_def)
149  for capture in collected_captures:
150  self.add_capture_arg(capture)
151  for use in collected_uses:
152  self.add_use(use)
153 
154  # Now normalize all defs and uses indexing maps now that full count of
155  # dims and symbols are known.
156  for cuse in self.uses.values():
157  cuse.indexing_map = self._normalize_affine_map(cuse.indexing_map)
158  for cdef in self.tensor_args.values():
159  cdef.shape_map = self._normalize_affine_map(cdef.shape_map,
160  with_dims=False)
161 
162  # Now for each write use, propagate the indexing maps from the use to the
163  # tensor, ensuring that there are not conflicts.
164  for write_use, _ in self.writes:
165  write_tensor_def = self.tensor_args[write_use.tensor_def]
166  if write_tensor_def.indexing_map:
167  raise ValueError(
168  f"Unexpected multi-write to a single tensor: {write_tensor_def}")
169  write_tensor_def.indexing_map = self.uses[write_use].indexing_map
170 
171  # For each read use, propagate the indexing maps from the use to the
172  # tensor, ensuring that there are not conflicts.
173  for _, read_expr in self.writes:
174  read_uses = set() # type: Set[TensorUse]
175  read_expr.collect_uses(read_uses)
176  for read_use in read_uses:
177  read_tensor_def = self.tensor_args[read_use.tensor_def]
178  if (read_tensor_def.indexing_map and
179  read_tensor_def.indexing_map != self.uses[read_use].indexing_map):
180  raise ValueError(
181  f"Unexpected multi-read of a tensor with different accesses:"
182  f"{read_tensor_def} vs {read_use}")
183  read_tensor_def.indexing_map = self.uses[read_use].indexing_map
184 
185  # Sanity check that all defs have an indexing map.
186  assert all(d.indexing_map for d in self.tensor_args.values()), (
187  f"Missing indexing map on TensorDef: {self.tensor_args}")
188 
189  # Collect reduction dims and ensure all the same.
190  all_reduction_dims = set(comprehension.all_reduction_dims)
191  if len(all_reduction_dims) != 1:
192  raise ValueError(
193  f"All writes within a generic must have the same reduction "
194  f"dims. Got: {all_reduction_dims}")
195  self.reduction_dims = next(iter(all_reduction_dims))
196 
197  # Check the index dimension exists and resolve
198  for index in collected_indices:
199  if index.dim_def.dimname not in self.affine_state.all_dims:
200  raise ValueError(
201  f"The dimension {index.dim.dimname} is not part of the iteration "
202  f"domain {self.affine_state.all_dims}")
203  index.resolve_dimension_name(self.affine_state)
204 
205  # Generate the scalar assignments (used to build a body).
206  self.assignments = [
207  ScalarAssign(write_use.tensor_name, read_expr.to_scalar_expression())
208  for write_use, read_expr in self.writes
209  ]

242 
243  def get_type(symbolic_name, position):
244  for reduction_dim_expr in self.reduction_dims:
245  if reduction_dim_expr.dimname == symbolic_name:
246  return "reduction"
247  return "parallel"

251  def add_tensor_arg(self, tensor_def: TensorDef):
252  if tensor_def in self.tensor_args:
253  return
254  with self.context:
255  local_state = AffineBuildState(global_state=self.affine_state,
256  allow_new_dims=False)
257  exprs = []
258  for expr in tensor_def.shape:
259  exprs.append(expr.build(state=local_state))
260  assert local_state.local_dim_count == 0
261  indexing_map = _ir.AffineMap.get(dim_count=0,
262  symbol_count=local_state.symbol_count,
263  exprs=exprs)
264 
265  def_config = TensorDefConfig(tensor_def, indexing_map)
266  self.tensor_args[tensor_def] = def_config
267 
268  def add_use(self, tensor_use: TensorUse):
269  if tensor_use in self.uses:
270  return
271  with self.context:
272  local_state = AffineBuildState(global_state=self.affine_state,
273  allow_new_symbols=False)
274  exprs = []
275  for expr in tensor_use.indices:
276  exprs.append(expr.build(state=local_state))
277  assert local_state.local_symbol_count == 0
278  indexing_map = _ir.AffineMap.get(dim_count=local_state.dim_count,
279  symbol_count=local_state.symbol_count,
280  exprs=exprs)
281 
282  use_config = TensorUseConfig(tensor_use, indexing_map)
283  self.uses[tensor_use] = use_config

285  def add_capture_arg(self, capture_def: CaptureDef):
286  if capture_def in self.capture_args:
