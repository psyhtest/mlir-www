100  @classmethod
101  def from_py_func(FuncOp,
102  *inputs: Type,
103  results: Optional[Sequence[Type]] = None,
104  name: Optional[str] = None):
105  """Decorator to define an MLIR FuncOp specified as a python function.
106 
107  Requires that an `mlir.ir.InsertionPoint` and `mlir.ir.Location` are
108  active for the current thread (i.e. established in a `with` block).
109 
110  When applied as a decorator to a Python function, an entry block will
111  be constructed for the FuncOp with types as specified in `*inputs`. The
112  block arguments will be passed positionally to the Python function. In
113  addition, if the Python function accepts keyword arguments generally or
114  has a corresponding keyword argument, the following will be passed:
115  * `func_op`: The `func` op being defined.
116 
117  By default, the function name will be the Python function `__name__`. This
118  can be overriden by passing the `name` argument to the decorator.
119 
120  If `results` is not specified, then the decorator will implicitly
121  insert a `ReturnOp` with the `Value`'s returned from the decorated
122  function. It will also set the `FuncOp` type with the actual return
123  value types. If `results` is specified, then the decorated function
124  must return `None` and no implicit `ReturnOp` is added (nor are the result
125  types updated). The implicit behavior is intended for simple, single-block
126  cases, and users should specify result types explicitly for any complicated
127  cases.
128 
129  The decorated function can further be called from Python and will insert
130  a `CallOp` at the then-current insertion point, returning either None (
131  if no return values), a unary Value (for one result), or a list of Values).
132  This mechanism cannot be used to emit recursive calls (by construction).
133  """
134 
135  def decorator(f):
136  from . import std
137  # Introspect the callable for optional features.
138  sig = inspect.signature(f)
139  has_arg_func_op = False
140  for param in sig.parameters.values():
141  if param.kind == param.VAR_KEYWORD:
142  has_arg_func_op = True
143  if param.name == "func_op" and (param.kind
144  == param.POSITIONAL_OR_KEYWORD or
145  param.kind == param.KEYWORD_ONLY):
146  has_arg_func_op = True
147 
148  # Emit the FuncOp.
149  implicit_return = results is None
150  symbol_name = name or f.__name__
151  function_type = FunctionType.get(
152  inputs=inputs, results=[] if implicit_return else results)
153  func_op = FuncOp(name=symbol_name, type=function_type)
154  with InsertionPoint(func_op.add_entry_block()):
155  func_args = func_op.entry_block.arguments
156  func_kwargs = {}
157  if has_arg_func_op:
158  func_kwargs["func_op"] = func_op
159  return_values = f(*func_args, **func_kwargs)
160  if not implicit_return:
161  return_types = list(results)
162  assert return_values is None, (
163  "Capturing a python function with explicit `results=` "
164  "requires that the wrapped function returns None.")
165  else:
166  # Coerce return values, add ReturnOp and rewrite func type.
167  if return_values is None:
168  return_values = []
169  elif isinstance(return_values, Value):
170  return_values = [return_values]
171  else:
172  return_values = list(return_values)
173  std.ReturnOp(return_values)
174  # Recompute the function type.
175  return_types = [v.type for v in return_values]
176  function_type = FunctionType.get(inputs=inputs, results=return_types)
177  func_op.attributes["type"] = TypeAttr.get(function_type)
178 
179  def emit_call_op(*call_args):
180  call_op = std.CallOp(return_types, FlatSymbolRefAttr.get(symbol_name),
181  call_args)
182  if return_types is None:
183  return None
184  elif len(return_types) == 1:
185  return call_op.result
186  else:
187  return call_op.results
188 
189  wrapped = emit_call_op
190  wrapped.__name__ = f.__name__
191  wrapped.func_op = func_op
192  return wrapped
